\section{凸集与凸函数}

\subsection{凸集的定义}

\begin{blackboard}[凸集：几何定义]

\begin{definition}[凸集]
集合 $C \subseteq \mathbb{R}^n$ 是\textbf{凸集}，当且仅当：

对任意 $x, y \in C$ 和任意 $\theta \in [0, 1]$，有 $\theta x + (1-\theta) y \in C$
\end{definition}

\textbf{直觉}：集合内任意两点的连线段都完全在集合内

\begin{center}
\begin{tikzpicture}[scale=1]

        \draw[red, thick] (-1, 0.3) -- (1, -0.2);
        \node[below] at (0, -1.5) {凸集 \checkmark};
    \end{scope}

    \end{scope}
\end{tikzpicture}
\end{center}

\end{blackboard}

\begin{blackboard}[常见的凸集]

\textbf{1. 超平面}：$\{x : a^T x = b\}$

\textbf{2. 半空间}：$\{x : a^T x \leq b\}$

\textbf{3. 多面体}（半空间的交集）：$\{x : Ax \leq b\}$

\textbf{4. 球}：$\{x : \|x - x_0\| \leq r\}$

\textbf{5. 椭球}：$\{x : (x - x_0)^T P^{-1} (x - x_0) \leq 1\}$，$P \succ 0$

\textbf{6. 锥}：$\{x : \|x\| \leq t\}$（二阶锥）

\textbf{7. 半正定锥}：$\{X \in \mathbb{R}^{n \times n} : X = X^T, X \succeq 0\}$

\textbf{关键性质}：凸集的交集仍是凸集

\end{blackboard}

\subsection{凸函数的定义}

\begin{blackboard}[凸函数：几何定义]

\begin{definition}[凸函数]
函数 $f: \mathbb{R}^n \to \mathbb{R}$ 是\textbf{凸函数}，当且仅当：

对任意 $x, y$ 和任意 $\theta \in [0, 1]$：
\[
f(\theta x + (1-\theta) y) \leq \theta f(x) + (1-\theta) f(y)
\]
\end{definition}

\textbf{直觉}：函数图像上任意两点连线在函数图像的\textbf{上方}

\begin{center}

\end{blackboard}

\begin{blackboard}[Jensen 不等式]

\begin{theorem}[Jensen 不等式]
如果 $f$ 是凸函数，$x_1, \ldots, x_k$ 是任意点，$\theta_1, \ldots, \theta_k \geq 0$ 且 $\sum_i \theta_i = 1$，则：
\[
f\left(\sum_{i=1}^k \theta_i x_i\right) \leq \sum_{i=1}^k \theta_i f(x_i)
\]
\end{theorem}

\textbf{直觉}：``加权平均后求函数值'' $\leq$ ``函数值的加权平均''

\textbf{特例}（期望形式）：如果 $X$ 是随机变量，$f$ 是凸函数：
\[
f(\mathbb{E}[X]) \leq \mathbb{E}[f(X)]
\]

\textbf{应用}：
\begin{itemize}
    \item $f(x) = x^2$ 凸 $\Rightarrow$ $(\mathbb{E}[X])^2 \leq \mathbb{E}[X^2]$，即 $\text{Var}(X) \geq 0$
    \item $f(x) = -\log x$ 凸 $\Rightarrow$ $-\log(\mathbb{E}[X]) \leq \mathbb{E}[-\log X]$（AM-GM 不等式）
\end{itemize}

\end{blackboard}

\begin{blackboard}[常见的凸函数]

\textbf{一元凸函数}：
\begin{itemize}
    \item $x^2, |x|, e^x, x \log x$ （$x > 0$），$-\log x$ （$x > 0$）
\end{itemize}

\textbf{多元凸函数}：
\begin{itemize}
    \item 仿射函数：$a^T x + b$（既凸又凹）
    \item 范数：$\|x\|_1, \|x\|_2, \|x\|_\infty$
    \item 二次函数：$x^T P x$，当 $P \succeq 0$
    \item Log-sum-exp：$\log(e^{x_1} + \cdots + e^{x_n})$
    \item 最大值函数：$\max\{x_1, \ldots, x_n\}$
\end{itemize}

\textbf{保持凸性的操作}：
\begin{itemize}
    \item 非负加权和：$\sum_i \alpha_i f_i$，$\alpha_i \geq 0$
    \item 复合仿射：$f(Ax + b)$
    \item 逐点最大：$\max\{f_1(x), f_2(x)\}$
\end{itemize}

\end{blackboard}

\section{凸性判定}
%========================================

\begin{blackboard}[如何判断函数是否凸？]

\textbf{方法1}：直接验证定义（通常很难）

\textbf{方法2}：一阶条件（需要可微）

\textbf{方法3}：二阶条件（需要二次可微）

\textbf{方法4}：证明它是凸函数的组合

\end{blackboard}

\subsection{一阶条件}

\begin{blackboard}[一阶条件：切线在下方]

\begin{theorem}[凸性的一阶条件]
设 $f$ 可微。$f$ 是凸函数当且仅当对所有 $x, y$：
\[
f(y) \geq f(x) + \nabla f(x)^T (y - x)
\]
\end{theorem}

\textbf{几何意义}：函数图像始终在其切平面的\textbf{上方}

\begin{center}
\begin{tikzpicture}[scale=0.9]
    \draw[->] (0, 0) -- (6, 0) node[right] {$x$};
    \draw[->] (0, 0) -- (0, 4) node[above] {$f(x)$};
    
    % 凸函数
    \draw[thick, blue, domain=0.5:5.5, samples=100] plot (\x, {0.15*(\x-3)*(\x-3) + 0.5});
    
    % 切线（在 x=2 处）
    \draw[red, thick, domain=0.5:5] plot (\x, {0.15*(2-3)*(2-3) + 0.5 + 0.3*(2-3)*(\x - 2)});
    
    \fill[red] (2, {0.15*(2-3)*(2-3) + 0.5}) circle (2pt);
    \node[below left] at (2, {0.15*(2-3)*(2-3) + 0.5}) {$(x, f(x))$};
    
    \node[right, red] at (5, {0.15*(2-3)*(2-3) + 0.5 + 0.3*(2-3)*(5 - 2)}) {切线};
    \node[right, blue] at (5.5, {0.15*(5.5-3)*(5.5-3) + 0.5}) {$f$};
\end{tikzpicture}
\end{center}

\textbf{推论}：如果 $\nabla f(x^*) = 0$，则 $x^*$ 是\textbf{全局最小值}！

\end{blackboard}

\begin{blackboard}[一阶条件的证明 $\bigstar$]

\textbf{证明}（$\Rightarrow$）：设 $f$ 凸，要证 $f(y) \geq f(x) + \nabla f(x)^T(y-x)$

由凸性定义，对 $\theta \in (0,1)$：
\[
f(x + \theta(y-x)) \leq (1-\theta) f(x) + \theta f(y)
\]

移项：
\[
f(x + \theta(y-x)) - f(x) \leq \theta (f(y) - f(x))
\]

除以 $\theta$：
\[
\frac{f(x + \theta(y-x)) - f(x)}{\theta} \leq f(y) - f(x)
\]

令 $\theta \to 0^+$，左边趋于方向导数 $\nabla f(x)^T(y-x)$：
\[
\nabla f(x)^T(y-x) \leq f(y) - f(x)
\]

即 $f(y) \geq f(x) + \nabla f(x)^T(y-x)$ \hfill $\square$

\end{blackboard}

\subsection{二阶条件}

\begin{blackboard}[二阶条件：Hessian 半正定]

\begin{theorem}[凸性的二阶条件]
设 $f$ 二次可微。$f$ 是凸函数当且仅当对所有 $x$：
\[
\nabla^2 f(x) \succeq 0
\]
即 Hessian 矩阵处处半正定。
\end{theorem}

\textbf{回忆}：矩阵 $H$ 半正定 $\Leftrightarrow$ 所有特征值 $\geq 0$ $\Leftrightarrow$ $v^T H v \geq 0$ 对所有 $v$

\textbf{直觉}：二阶导数处处非负 $\Rightarrow$ 函数``向上弯曲''

\end{blackboard}

\begin{blackboard}[二阶条件的证明 $\bigstar$]

\textbf{证明}（$\Rightarrow$）：设 $f$ 凸，要证 $\nabla^2 f(x) \succeq 0$

由一阶条件，对任意 $y$：
\[
f(y) \geq f(x) + \nabla f(x)^T(y-x)
\]

同时，交换 $x, y$ 的角色：
\[
f(x) \geq f(y) + \nabla f(y)^T(x-y)
\]

两式相加：
\[
0 \geq (\nabla f(x) - \nabla f(y))^T(y-x) = -(\nabla f(y) - \nabla f(x))^T(y-x)
\]

即 $(\nabla f(y) - \nabla f(x))^T(y-x) \geq 0$（梯度是\textbf{单调}的）

设 $y = x + tv$，$t > 0$：
\[
(\nabla f(x+tv) - \nabla f(x))^T v \geq 0
\]

除以 $t$，令 $t \to 0$：
\[
v^T \nabla^2 f(x) v \geq 0
\]

对所有 $v$ 成立，故 $\nabla^2 f(x) \succeq 0$ \hfill $\square$

\end{blackboard}

\begin{blackboard}[例子：二次函数]

\textbf{函数}：$f(x) = \frac{1}{2} x^T P x + q^T x + r$

\textbf{梯度}：$\nabla f(x) = Px + q$

\textbf{Hessian}：$\nabla^2 f(x) = P$

\textbf{凸性}：$f$ 是凸函数 $\Leftrightarrow$ $P \succeq 0$

\textbf{强凸性}：$f$ 是强凸函数 $\Leftrightarrow$ $P \succ 0$

\end{blackboard}

\begin{blackboard}[强凸性]

\begin{definition}[强凸函数]
函数 $f$ 是 \textbf{$m$-强凸}的，如果对所有 $x$：
\[
\nabla^2 f(x) \succeq m I, \quad m > 0
\]
\end{definition}

\textbf{等价定义}：$f(x) - \frac{m}{2}\|x\|^2$ 是凸函数

\textbf{性质}：
\begin{itemize}
    \item 强凸函数有\textbf{唯一}的全局最小值
    \item 梯度下降收敛更快（线性收敛 vs 次线性收敛）
\end{itemize}

\section{凸的礼物：局部即全局}
%========================================

\begin{blackboard}[凸优化的核心定理]

\begin{theorem}[局部最优 = 全局最优]
对于凸优化问题：
\[
\min_x f(x) \quad \text{s.t.} \quad x \in C
\]
其中 $f$ 是凸函数，$C$ 是凸集。

如果 $x^*$ 是\textbf{局部最优解}，则 $x^*$ 是\textbf{全局最优解}。
\end{theorem}

\textbf{证明}：反证法。假设存在 $y \in C$ 使得 $f(y) < f(x^*)$。

由凸性，对任意 $\theta \in (0, 1)$：
\[
f(\theta y + (1-\theta) x^*) \leq \theta f(y) + (1-\theta) f(x^*) < f(x^*)
\]

当 $\theta \to 0$ 时，$\theta y + (1-\theta) x^* \to x^*$，矛盾于 $x^*$ 是局部最优。

\end{blackboard}

\begin{insight}[为什么这个定理如此重要？]

\textbf{非凸优化的噩梦}：
\begin{itemize}
    \item 梯度下降可能陷入局部最优
    \item 不知道找到的解有多好
    \item 可能需要多次随机重启
\end{itemize}

\textbf{凸优化的美梦}：
\begin{itemize}
    \item 任何下降方法都能找到全局最优
    \item 有收敛性保证
    \item 一次运行就够
\end{itemize}

\textbf{结论}：证明问题是凸的，是优化研究的第一步！

\end{insight}

\section{梯度下降的收敛性 $\bigstar$}
%========================================

\begin{blackboard}[基本假设与记号]

设 $f: \mathbb{R}^n \to \mathbb{R}$ 是可微函数，$x^* = \arg\min_x f(x)$ 是全局最优解，$f^* = f(x^*)$。

\textbf{梯度下降迭代}（步长 $\alpha > 0$）：
\[
    x_{t+1} = x_t - \alpha \nabla f(x_t)
\]

本节使用以下两个核心假设：

\begin{enumerate}
    \item \textbf{凸性}：$f$ 是凸函数，即对所有 $x, y$：
    \[
        f(y) \geq f(x) + \nabla f(x)^T (y - x)
    \]
    \item \textbf{$L$-光滑性}（梯度 Lipschitz 连续）：存在常数 $L > 0$，使得对所有 $x, y$：
    \[
        \|\nabla f(x) - \nabla f(y)\|_2 \leq L \|x - y\|_2
    \]
    等价地，对所有 $x, y$：
    \[
        f(y) \leq f(x) + \nabla f(x)^T(y - x) + \frac{L}{2}\|y - x\|^2
        \quad \text{（下降引理）}
    \]
\end{enumerate}

\textbf{直觉}：$L$-光滑意味着函数不会"弯曲得太快"，梯度信息在一步之内仍然可靠。

\end{blackboard}

\begin{blackboard}[关键引理]

\begin{lemma}[充分下降引理]
设 $f$ 是 $L$-光滑函数，步长 $\alpha = 1/L$，则梯度下降每步保证：
\[
    f(x_{t+1}) \leq f(x_t) - \frac{1}{2L}\|\nabla f(x_t)\|^2
\]
\end{lemma}

\textbf{证明}：将 $y = x_{t+1} = x_t - \frac{1}{L}\nabla f(x_t)$ 代入下降引理：
\[
    f(x_{t+1}) \leq f(x_t) + \nabla f(x_t)^T\!\left(-\frac{1}{L}\nabla f(x_t)\right)
    + \frac{L}{2}\left\|\frac{1}{L}\nabla f(x_t)\right\|^2
    = f(x_t) - \frac{1}{2L}\|\nabla f(x_t)\|^2 \qquad \square
\]

\textbf{推论}：$f(x_t)$ 单调不增，即梯度下降在凸 $L$-光滑函数上\textbf{永远不会变差}。

\end{blackboard}

\begin{blackboard}[主定理：$O(1/T)$ 收敛率]

\begin{theorem}[凸函数梯度下降收敛率]
设 $f$ 是凸函数且 $L$-光滑，步长 $\alpha = 1/L$。则对任意 $T \geq 1$：
\[
    f(x_T) - f^* \leq \frac{L\,\|x_0 - x^*\|^2}{2T}
\]
\end{theorem}

\textbf{证明}：

\textbf{第一步}：建立距离递推关系。令 $r_t = \|x_t - x^*\|^2$，展开：
\begin{align*}
    r_{t+1} &= \left\|x_t - \frac{1}{L}\nabla f(x_t) - x^*\right\|^2 \\
    &= r_t - \frac{2}{L}\nabla f(x_t)^T(x_t - x^*) + \frac{1}{L^2}\|\nabla f(x_t)\|^2
\end{align*}

\textbf{第二步}：利用凸性下界。由凸性一阶条件（取 $y = x^*$）：
\[
    f(x_t) - f^* \leq \nabla f(x_t)^T(x_t - x^*)
\]

令 $\delta_t = f(x_t) - f^*$，代入第一步：
\[
    r_{t+1} \leq r_t - \frac{2}{L}\delta_t + \frac{1}{L^2}\|\nabla f(x_t)\|^2
\]

\textbf{第三步}：利用充分下降引理消去梯度范数。由引理知 $\|\nabla f(x_t)\|^2 \leq 2L(\delta_t - \delta_{t+1})$，故：
\[
    r_{t+1} \leq r_t - \frac{2}{L}\delta_t + \frac{2}{L}(\delta_t - \delta_{t+1})
    = r_t - \frac{2}{L}\delta_{t+1}
\]

整理得：
\[
    \frac{2}{L}\delta_{t+1} \leq r_t - r_{t+1}
\]

\textbf{第四步}：对 $t = 0, 1, \ldots, T-1$ 求和（望远镜求和）：
\[
    \frac{2}{L}\sum_{t=1}^{T}\delta_t \leq r_0 - r_T \leq r_0 = \|x_0 - x^*\|^2
\]

\textbf{第五步}：由充分下降引理知 $\delta_t$ 单调不增，故 $T \cdot \delta_T \leq \sum_{t=1}^T \delta_t$，因此：
\[
    \frac{2T}{L}\delta_T \leq \|x_0 - x^*\|^2
    \implies \boxed{f(x_T) - f^* \leq \frac{L\,\|x_0 - x^*\|^2}{2T}} \qquad \square
\]

\end{blackboard}

\begin{keypoint}[$O(1/T)$ 收敛率的含义]

\begin{itemize}
    \item 要使误差 $\leq \varepsilon$，需要迭代次数 $T = O(L \|x_0 - x^*\|^2 / \varepsilon)$，即 $O(1/\varepsilon)$ 步。
    \item 每步计算量为 $O(n)$（只算一次梯度），总计算量 $O(n/\varepsilon)$。
    \item 这被称为\textbf{次线性收敛}（sublinear convergence）：误差以 $1/T$ 的速度衰减。
    \item 此速率对一般凸函数是\textbf{最优的}（信息论下界），即无法在相同假设下做得更好。
\end{itemize}

\end{keypoint}

\begin{blackboard}[强化：强凸函数的线性收敛率 $\bigstar$]

若进一步假设 $f$ 是 \textbf{$\mu$-强凸}（$\mu > 0$），即对所有 $x, y$：
\[
    f(y) \geq f(x) + \nabla f(x)^T(y-x) + \frac{\mu}{2}\|y-x\|^2
\]

\begin{theorem}[强凸函数线性收敛率]
设 $f$ 是 $\mu$-强凸且 $L$-光滑，步长 $\alpha = 1/L$，则：
\[
    \|x_T - x^*\|^2 \leq \left(1 - \frac{\mu}{L}\right)^T \|x_0 - x^*\|^2
\]
\end{theorem}

\textbf{证明}：

由强凸性（令 $y = x^*$）：
\[
    f(x_t) - f^* \geq \frac{\mu}{2}\|x_t - x^*\|^2 = \frac{\mu}{2} r_t
\]

由第三步的结论 $r_{t+1} \leq r_t - \frac{2}{L}\delta_{t+1}$，利用充分下降 $\delta_{t+1} \geq \frac{\mu}{2}r_{t+1}$：

实际上，更直接地，从第一步和强凸性下界出发：
\[
    r_{t+1} \leq r_t - \frac{2}{L}\delta_t + \frac{1}{L^2}\|\nabla f(x_t)\|^2
\]

由强凸性推出 $\|\nabla f(x_t)\|^2 \geq 2\mu \delta_t$（证明略），结合 $\frac{1}{L^2}\|\nabla f(x_t)\|^2 \leq \frac{2}{L}\delta_t - \frac{2}{L}\delta_{t+1}$，

更简洁地，利用强凸性的等价形式（强单调性）直接得：
\begin{align*}
    r_{t+1} &= \left\|x_t - \frac{1}{L}\nabla f(x_t) - x^*\right\|^2 \\
    &= r_t - \frac{2}{L}\underbrace{\nabla f(x_t)^T(x_t - x^*)}_{\geq \frac{\mu}{2}r_t + \frac{1}{2L}\|\nabla f(x_t)\|^2}
    + \frac{1}{L^2}\|\nabla f(x_t)\|^2 \\
    &\leq r_t - \frac{\mu}{L} r_t - \frac{1}{L^2}\|\nabla f(x_t)\|^2 + \frac{1}{L^2}\|\nabla f(x_t)\|^2 \\
    &= \left(1 - \frac{\mu}{L}\right) r_t
\end{align*}

（第二行利用了强凸性的等价不等式 $\nabla f(x)^T(x-y) \geq \frac{\mu}{2}\|x-y\|^2 + \frac{1}{2L}\|\nabla f(x) - \nabla f(y)\|^2$，取 $y=x^*$ 且 $\nabla f(x^*)=0$。）

递归展开得：
\[
    \boxed{\|x_T - x^*\|^2 \leq \left(1 - \frac{\mu}{L}\right)^T \|x_0 - x^*\|^2} \qquad \square
\]

\end{blackboard}

\begin{keypoint}[线性收敛率的含义]

\begin{itemize}
    \item 定义\textbf{条件数} $\kappa = L/\mu \geq 1$，收缩因子为 $1 - 1/\kappa$。
    \item 要使 $\|x_T - x^*\|^2 \leq \varepsilon$，需要 $T = O(\kappa \log(1/\varepsilon))$ 步，即 $O(\log(1/\varepsilon))$ 步——远优于 $O(1/\varepsilon)$！
    \item 这是\textbf{线性收敛}（linear convergence）：误差的对数随迭代线性下降，即"每次迭代误差减少一个固定百分比"。
    \item $\kappa$ 越大（病态程度越高），收敛越慢；$\kappa = 1$ 时（即 $\mu = L$）一步到达最优。
\end{itemize}

\begin{center}
\begin{tabular}{c|c|c|c}
假设 & 收敛率 & 步数达到精度 $\varepsilon$ & 名称 \\
\hline
凸 $+$ $L$-光滑 & $O(1/T)$ & $O(1/\varepsilon)$ & 次线性收敛 \\
强凸 $+$ $L$-光滑 & $O((1-\mu/L)^T)$ & $O(\kappa \log 1/\varepsilon)$ & 线性收敛
\end{tabular}
\end{center}

\end{keypoint}

\begin{example}[具体例子：$f(x) = kx^2$ 的梯度下降]

取 $f(x) = kx^2$，$k > 0$，最优解显然是 $x^* = 0$，$f^* = 0$。

\textbf{第一步：确定 $L$ 和 $\mu$}

$f''(x) = 2k$（常数），故 $f$ 同时是 $L = 2k$-光滑的和 $\mu = 2k$-强凸的，条件数 $\kappa = L/\mu = 1$。

\textbf{第二步：写出迭代格式}

$\nabla f(x) = 2kx$，梯度下降：
\[
x_{t+1} = x_t - \alpha \cdot 2k x_t = (1 - 2k\alpha)\, x_t
\]

记收缩因子 $\rho = 1 - 2k\alpha$，递归展开得\textbf{闭合公式}：
\[
\boxed{x_t = \rho^t\, x_0 = (1 - 2k\alpha)^t\, x_0}
\]

\textbf{第三步：分析不同步长的行为}

\begin{center}
\begin{tabular}{c|c|c|l}
步长 $\alpha$ & $\rho = 1-2k\alpha$ & $|\rho|$ & 行为 \\
\hline
$\dfrac{1}{2k}$（$= 1/L$，最优） & $0$ & $0$ & \textbf{一步收敛}！$x_1 = 0$ \\[6pt]
$\dfrac{1}{4k}$ & $\dfrac{1}{2}$ & $0.5$ & 单调收敛，每步误差减半 \\[6pt]
$\dfrac{3}{4k}$ & $-\dfrac{1}{2}$ & $0.5$ & 振荡收敛（正负交替，幅度减半） \\[6pt]
$\dfrac{1}{k}$（$= 2/L$，临界） & $-1$ & $1$ & 永远振荡，不收敛 \\[6pt]
$> \dfrac{1}{k}$ & $|\rho|>1$ & $>1$ & 发散（步长太大）
\end{tabular}
\end{center}

\textbf{直觉图}（$k=1$，$x_0 = 1$，不同步长的轨迹）：

\textbf{第四步：与一般理论对照}

\begin{itemize}
    \item 最优步长 $\alpha = 1/L = 1/(2k)$，此时 $\rho = 0$——$\kappa = 1$ 的情形理论预测 $(1-\mu/L)^T = 0^T = 0$，即一步精确到达，与计算一致。
    \item 步长约束 $0 < \alpha < 1/k = 2/L$ 与充分下降引理的要求 $\alpha \leq 1/L$ 对应（$\alpha \leq 1/L$ 保证单调下降，$1/L < \alpha < 2/L$ 振荡但仍收敛，$\alpha \geq 2/L$ 发散）。
    \item 误差公式 $|x_t - x^*| = |\rho|^t |x_0|$ 正是线性收敛 $O((1-\mu/L)^T)$ 的具体体现。
\end{itemize}

\end{example}

\begin{insight}[为什么凸性保证梯度下降找到最优解？]

核心逻辑链：

\begin{enumerate}
    \item \textbf{凸性 $\Rightarrow$ 局部 = 全局}：不存在"陷阱"式局部最优。
    \item \textbf{$L$-光滑 $\Rightarrow$ 每步必下降}（充分下降引理）：梯度下降永远在改善。
    \item \textbf{两者合并 $\Rightarrow$ 收敛到全局最优}：既然每步都在改善，且改善终点就是全局最优，梯度下降必然收敛至全局最优。
\end{enumerate}

对比非凸情形：满足下降引理，但可能收敛到局部最优——有下降保证，却无全局保证。

\end{insight}

\section{凸优化的标准形式}
%========================================

\begin{insight}[既然梯度下降能解无约束凸优化，为什么还要搞有约束的？]

\textbf{问题}：上一节证明了，对无约束凸优化 $\min_x f(x)$，梯度下降必然收敛到全局最优。那为什么不把所有问题都写成无约束形式，直接梯度下降了事？

\textbf{原因一：现实问题本身就有约束}

很多约束无法"消掉"，它们是问题的本质：
\begin{itemize}
    \item \textbf{资源限制}：预算 $\leq$ 100 万、时间 $\leq$ 24 小时、库存 $\geq$ 0
    \item \textbf{物理约束}：功率 $\leq$ 额定值、速度 $\leq$ 光速
    \item \textbf{规则约束}：概率之和 $= 1$、流量守恒等式
    \item \textbf{安全约束}：距离 $\geq$ 安全距离、应力 $\leq$ 材料极限
\end{itemize}

\textbf{原因二：硬约束 $\neq$ 罚函数}

一种"逃脱"约束的思路是把约束加到目标函数里（罚函数法）：
\[
    \min_x f(x) + \rho \cdot \max(0,\, g(x))
\]
但这只是近似——罚函数法找到的解\textbf{一般不严格满足约束}，$\rho$ 调多大都只是逼近。而有些场景（安全、合规、物理可行性）要求约束\textbf{精确成立}，差一点都不行。

\textbf{原因三：约束改变了最优解的位置}

\begin{center}
\begin{tikzpicture}[scale=1.1]
    % 等高线
    \foreach \r in {0.5, 1.0, 1.5, 2.0} {
        \draw[blue!40, thin] (0,0) ellipse ({\r*1.5} and {\r*0.8});
    }
    % 无约束最优
    \fill[blue] (0,0) circle (2.5pt) node[below right, font=\small] {无约束最优 $x^*$};
    % 可行域
    \fill[green!20, opacity=0.6] (1.0,-1.5) rectangle (3.5, 1.5);
    \draw[green!60!black, thick] (1.0,-1.5) -- (1.0, 1.5);
    \node[green!60!black, font=\small] at (2.3, 0) {可行域};
    % 约束最优
    \fill[red] (1.0, 0) circle (2.5pt) node[above right, font=\small] {约束最优 $x^*_c$};
    % 箭头
    \draw[->, dashed, thick, myred] (0,0) -- (0.85, 0);
\end{tikzpicture}
\end{center}

无约束最优 $x^*$ 在可行域\textbf{之外}，梯度下降"直奔"$x^*$ 根本不会停在可行域边界上，给出的解是不可行的。

\textbf{结论}：有约束凸优化 $\neq$ 无约束凸优化。我们需要专门的理论（Lagrangian、KKT 条件）来刻画约束最优解的结构，以及专门的算法（投影梯度下降、内点法、ADMM）来高效求解。

\end{insight}

\begin{blackboard}[凸优化的标准形式]

\textbf{标准形式}：
\[
\begin{aligned}
\min_{x} \quad & f(x) \\
\text{s.t.} \quad & g_i(x) \leq 0, \quad i = 1, \ldots, m \\
& h_j(x) = 0, \quad j = 1, \ldots, p
\end{aligned}
\]

\textbf{凸优化的要求}：
\begin{itemize}
    \item 目标函数 $f$ 是\textbf{凸函数}
    \item 不等式约束 $g_i$ 是\textbf{凸函数}
    \item 等式约束 $h_j$ 是\textbf{仿射函数}（$h_j(x) = a_j^T x - b_j$）
\end{itemize}

\textbf{注意}：等式约束只能是仿射的！

\textbf{原因}：$h(x) = 0$ 等价于 $h(x) \leq 0$ 且 $-h(x) \leq 0$，要同时凸只能是仿射

\end{blackboard}

\begin{blackboard}[常见的凸优化问题]

\textbf{1. 线性规划 (LP)}：
\[
\min_x c^T x \quad \text{s.t.} \quad Ax \leq b, \quad Cx = d
\]

\textbf{2. 二次规划 (QP)}：
\[
\min_x \frac{1}{2} x^T P x + q^T x \quad \text{s.t.} \quad Ax \leq b
\]
（$P \succeq 0$）

\textbf{3. 二阶锥规划 (SOCP)}：
\[
\min_x c^T x \quad \text{s.t.} \quad \|A_i x + b_i\|_2 \leq c_i^T x + d_i
\]

\textbf{4. 半定规划 (SDP)}：
\[
\min_X \langle C, X \rangle \quad \text{s.t.} \quad \langle A_i, X \rangle = b_i, \quad X \succeq 0
\]

\textbf{包含关系}：LP $\subset$ QP $\subset$ SOCP $\subset$ SDP $\subset$ 凸优化

\end{blackboard}

\begin{blackboard}[单纯形法（Simplex Method）]

\textbf{几何直觉}：LP 的最优解必然在可行多面体的\textbf{顶点}（极点）上。

\begin{center}
\begin{tikzpicture}[scale=0.85]
    % 可行域
    \fill[blue!10] (0,0) -- (4,0) -- (3,2) -- (1,3) -- (0,2) -- cycle;
    \draw[thick] (0,0) -- (4,0) -- (3,2) -- (1,3) -- (0,2) -- cycle;
    % 顶点
    \foreach \p in {(0,0),(4,0),(3,2),(1,3),(0,2)} { \fill[blue] \p circle (2.5pt); }
    % 目标函数等高线
    \draw[red, dashed] (-0.2, 2.6) -- (3.5, 0.5) node[right, font=\small] {$c^Tx = $ const};
    \draw[red, dashed] (-0.2, 3.6) -- (4, 1.0);
    % 最优顶点
    \fill[red] (1,3) circle (3pt) node[above right] {最优顶点};
    % 路径
    \draw[->, thick, mygreen] (0,0) -- (4,0);
    \draw[->, thick, mygreen] (4,0) -- (3,2);
    \draw[->, thick, mygreen] (3,2) -- (1,3);
    \node[mygreen, font=\small] at (2.5,-0.4) {单纯形路径};
\end{tikzpicture}
\end{center}

\textbf{算法框架}：

\begin{enumerate}
    \item \textbf{初始化}：找一个初始顶点（基可行解，BFS）。
    \item \textbf{检验}：计算相邻顶点的目标值是否有改善（用\textbf{简约费用} $\bar{c}_j = c_j - c_B^T B^{-1} A_j$）。
    \item \textbf{转轴}（Pivot）：若存在 $\bar{c}_j < 0$（最小化问题），沿该方向移动到相邻顶点（\textbf{换基}）。
    \item \textbf{终止}：所有 $\bar{c}_j \geq 0$，当前顶点即最优。
\end{enumerate}

\textbf{每步操作}：更新基矩阵 $B$，求解 $B^{-1}$——每步代价 $O(mn)$（利用上次迭代的 LU 分解热启动）。

\textbf{复杂度分析}：

\begin{itemize}
    \item \textbf{每步}：$O(mn)$
    \item \textbf{步数（最坏）}：$O\!\left(\binom{n}{m}\right) = 2^{\Omega(n)}$（Klee-Minty 例子），理论上指数级
    \item \textbf{步数（实践）}：通常 $O(m)$ 到 $O(3m)$ 步，实践中极快
    \item \textbf{总体}：理论 NP（最坏），实践近似 $O(m^2 n)$（多项式行为）
\end{itemize}

\begin{keypoint}[单纯形法的悖论]
最坏情况指数，但五十年来工程实践无敌。原因：实际问题的多面体"看起来"像球体（顶点数少、直径小），随机旋转规则下期望步数为 $O(m)$。
\end{keypoint}

\end{blackboard}

\begin{blackboard}[内点法（Interior Point Method / Barrier Method）]

\textbf{几何直觉}：不走顶点，而是走\textbf{可行域内部}，沿中心路径（central path）趋向最优点。

\begin{center}
\begin{tikzpicture}[scale=0.85]
    \fill[blue!10] (0,0) -- (4,0) -- (3,2) -- (1,3) -- (0,2) -- cycle;
    \draw[thick] (0,0) -- (4,0) -- (3,2) -- (1,3) -- (0,2) -- cycle;
    \foreach \p in {(0,0),(4,0),(3,2),(1,3),(0,2)} { \fill[blue] \p circle (2.5pt); }
    % 中心路径
    \draw[red, thick, ->] (1.5,1.5) .. controls (1.3,2.0) and (1.1,2.5) .. (1,3);
    \fill[red] (1.5,1.5) circle (2.5pt);
    \fill[red] (1,3) circle (3pt) node[above right] {最优点};
    \node[red, font=\small] at (2.8, 1.8) {中心路径};
\end{tikzpicture}
\end{center}

\textbf{核心思路}：用对数障碍函数（log-barrier）将约束``软化''到目标里：
\[
\min_x \; c^T x - \frac{1}{t} \sum_{i=1}^m \ln(-g_i(x))
\]
随参数 $t \to \infty$，障碍项消失，解趋向原始最优解。

每个固定 $t$ 下用 \textbf{Newton 法}求解（利用 Hessian 二阶信息），每步代价 $O(n^3)$（求解 $n\times n$ 线性方程组）或稀疏结构下更快。

\textbf{复杂度分析}：

\begin{itemize}
    \item \textbf{每步（Newton 迭代）}：$O(n^3)$（稠密）或 $O(n^{1.5})$（稀疏/结构化）
    \item \textbf{总步数}：$O(\sqrt{m} \log(1/\varepsilon))$——这是\textbf{多项式时间}！
    \item \textbf{总体}：$O(n^3 \sqrt{m} \log(1/\varepsilon))$，理论上优于单纯形
\end{itemize}

$O(\sqrt{m})$ 步的来源：每次 $t$ 增大一个常数倍，中心路径的总"弯折程度"（自协调参数）为 $O(\sqrt{m})$。

\textbf{对比总结}：

\begin{center}
\begin{tabular}{c|c|c|c}
 & 单纯形法 & 内点法 & 说明 \\
\hline
每步代价 & $O(mn)$ & $O(n^3)$ 或 $O(n^{1.5})$ & 内点法每步更贵 \\
步数（最坏） & 指数 & $O(\sqrt{m}\log 1/\varepsilon)$ & 内点法有理论保证 \\
步数（实践） & $O(m)$ & $O(30\text{--}100)$ & 内点法步数几乎与规模无关 \\
适合规模 & 中小型，稀疏 & 中等密集型 & 各有适用场景 \\
热启动 & 容易 & 困难 & 重复求解时单纯形更快 \\
解的性质 & 顶点解（稀疏） & 内部解（稠密） & 影响后续处理 \\
\end{tabular}
\end{center}

\end{blackboard}

\begin{insight}[为什么 LP 有多项式算法但还在用指数的单纯形法？]

\begin{itemize}
    \item \textbf{理论}：内点法 $O(n^{3.5} \log 1/\varepsilon)$ 是多项式时间，1984 年 Karmarkar 算法证明 LP $\in$ P。
    \item \textbf{实践}：商业求解器（Gurobi、CPLEX）默认单纯形法，原因：
    \begin{enumerate}
        \item 单纯形法的每步代价 $O(mn)$ 远小于内点法 $O(n^3)$
        \item 单纯形法解为顶点解（稀疏），便于热启动和整数规划
        \item 对大多数实际 LP，单纯形法 $O(m)$ 步的实践表现无敌
    \end{enumerate}
    \item \textbf{规模极大时}：一阶方法（投影梯度下降、ADMM）取代两者，每步 $O(mn)$ 但无需因子分解。
\end{itemize}

\end{insight}

\begin{blackboard}[QP 例子 1：Markowitz 投资组合]

\textbf{场景}：投资 $n$ 只股票，平衡收益与风险

\begin{itemize}
    \item $\mu_i$：股票 $i$ 的预期收益率
    \item $\Sigma$：收益率的协方差矩阵（风险）
    \item $x_i$：投资在股票 $i$ 的比例
\end{itemize}

\textbf{建模}：
\[
\begin{aligned}
\min_x \quad & \frac{1}{2} x^T \Sigma x \quad \text{（最小化风险）} \\
\text{s.t.} \quad & \mu^T x \geq r_{\min} \quad \text{（保证最低收益）} \\
& \mathbf{1}^T x = 1 \quad \text{（投资比例和为 1）} \\
& x \geq 0 \quad \text{（不允许卖空）}
\end{aligned}
\]

\textbf{有效前沿}：改变 $r_{\min}$，得到收益-风险权衡曲线

\end{blackboard}

\begin{blackboard}[QP 例子 2：支持向量机 (SVM)]

\textbf{场景}：二分类问题，找最大间隔超平面

\textbf{原始问题}：
\[
\begin{aligned}
\min_{w, b} \quad & \frac{1}{2} \|w\|^2 \\
\text{s.t.} \quad & y_i (w^T x_i + b) \geq 1, \quad \forall i
\end{aligned}
\]

这是一个 QP！目标函数是 $w$ 的二次函数，约束是线性的。

\textbf{软间隔 SVM}（允许误分类）：
\[
\begin{aligned}
\min_{w, b, \xi} \quad & \frac{1}{2} \|w\|^2 + C \sum_i \xi_i \\
\text{s.t.} \quad & y_i (w^T x_i + b) \geq 1 - \xi_i \\
& \xi_i \geq 0
\end{aligned}
\]

\end{blackboard}

\begin{blackboard}[QP 例子 3：岭回归]

\textbf{场景}：带正则化的线性回归

\textbf{普通最小二乘}：$\min_w \|Xw - y\|^2$

\textbf{问题}：当特征多、样本少时，过拟合严重

\textbf{岭回归}：
\[
\min_w \|Xw - y\|^2 + \lambda \|w\|^2
\]

展开：
\[
\min_w w^T (X^T X + \lambda I) w - 2 y^T X w + \text{const}
\]

这是 QP！$P = 2(X^T X + \lambda I) \succ 0$（强凸）

\textbf{闭式解}：$w^* = (X^T X + \lambda I)^{-1} X^T y$

\end{blackboard}

\subsection{二阶锥规划 (SOCP) 的例子}
%----------------------------------------

\begin{blackboard}[SOCP 例子 1：鲁棒线性规划]

\textbf{场景}：参数有不确定性的 LP

普通 LP：$\min c^T x$ s.t. $a_i^T x \leq b_i$

\textbf{问题}：如果 $a_i$ 有误差怎么办？

假设 $a_i \in \mathcal{E}_i = \{\bar{a}_i + P_i u : \|u\| \leq 1\}$（椭球不确定集）

\textbf{鲁棒约束}：对所有可能的 $a_i$ 都满足
\[
\max_{a_i \in \mathcal{E}_i} a_i^T x \leq b_i
\]

\textbf{等价于}：
\[
\bar{a}_i^T x + \|P_i^T x\| \leq b_i
\]

这是二阶锥约束！

\end{blackboard}

\begin{blackboard}[SOCP 例子 2：设施选址]

\textbf{场景}：在平面上选一个位置建设施，最小化到所有客户的总距离

客户位置：$(a_1, b_1), \ldots, (a_m, b_m)$

\textbf{建模}：设设施位置为 $(x, y)$
\[
\min_{x, y} \sum_{i=1}^m \sqrt{(x - a_i)^2 + (y - b_i)^2}
\]

\textbf{引入辅助变量} $t_i$：
\[
\begin{aligned}
\min \quad & \sum_i t_i \\
\text{s.t.} \quad & \|(x - a_i, y - b_i)\|_2 \leq t_i, \quad \forall i
\end{aligned}
\]

这是 SOCP！

\end{blackboard}

\begin{blackboard}[SOCP 例子 3：机器人路径规划]

\textbf{场景}：机器人从 $A$ 到 $B$，避开圆形障碍物

障碍物：圆心 $c_i$，半径 $r_i$

\textbf{约束}：路径点 $x_k$ 必须在所有障碍物外
\[
\|x_k - c_i\| \geq r_i + \epsilon
\]

\textbf{问题}：这是非凸的！（集合外部不是凸集）

\textbf{解决方案}：
\begin{itemize}
    \item 方法1：凸松弛 + 序列凸规划 (SCP)
    \item 方法2：整数规划（选择绕行方向）
    \item 方法3：采样方法（RRT）
\end{itemize}

\textbf{教训}：不是所有几何问题都能直接凸化

\end{blackboard}

\subsection{半定规划 (SDP) 的例子}
%----------------------------------------

\begin{blackboard}[SDP 例子 1：MAX-CUT 松弛]

\textbf{MAX-CUT 问题}：把图 $G=(V,E)$ 的顶点分成两组 $S$ 和 $\bar{S}$，最大化跨组的边数（或加权边权之和）。

\textbf{具体例子}：4个顶点、5条边的图

\begin{center}
\begin{tikzpicture}[scale=1]
    \node[draw, circle, fill=red!30] (1) at (0, 1) {1};
    \node[draw, circle, fill=red!30] (2) at (2, 1) {2};
    \node[draw, circle, fill=blue!30] (3) at (2, -1) {3};
    \node[draw, circle, fill=blue!30] (4) at (0, -1) {4};

    \draw[thick] (1) -- (2) node[midway, above] {1};
    \draw[thick] (2) -- (3) node[midway, right] {1};
    \draw[thick] (3) -- (4) node[midway, below] {1};
    \draw[thick] (4) -- (1) node[midway, left] {1};
    \draw[thick] (1) -- (3) node[midway, above right] {1};

    \node at (5, 0) {切割 $\{1,2\}$ vs $\{3,4\}$};
    \node at (5, -0.6) {跨组边数 = 4};
\end{tikzpicture}
\end{center}

\textbf{第一步：整数规划建模}

令 $x_i \in \{-1, +1\}$：$+1$ 表示顶点 $i$ 在 $S$，$-1$ 表示在 $\bar{S}$。

边 $(i,j)$ 被切割 $\Leftrightarrow$ $x_i \neq x_j$ $\Leftrightarrow$ $x_ix_j = -1$ $\Leftrightarrow$ $\frac{1-x_ix_j}{2} = 1$。

原问题（NP-Hard 整数规划）：
\[
\max_{x_i \in \{-1,+1\}} \sum_{(i,j)\in E} \frac{1-x_ix_j}{2}
\]

\textbf{第二步：SDP 松弛}

关键观察：$x_ix_j$（标量乘积）$\longrightarrow$ $v_i^Tv_j$（向量内积），把每个 $x_i \in \{-1,+1\}$ 替换为单位向量 $v_i \in \mathbb{R}^n$，$\|v_i\|_2 = 1$。

定义矩阵 $X_{ij} = v_i^Tv_j$（Gram 矩阵）。Gram 矩阵天然满足：
\begin{itemize}
    \item $X \succeq 0$（半正定）：因为 $u^TXu = \|\sum_i u_i v_i\|^2 \geq 0$
    \item $X_{ii} = v_i^Tv_i = \|v_i\|^2 = 1$（对角元为 1）
\end{itemize}

SDP 松弛（以 $X$ 为变量，$X \in \mathbb{R}^{4\times 4}$）：
\[
\max_X \; \frac{1}{2}\sum_{(i,j)\in E}(1 - X_{ij})
\quad \text{s.t.} \quad X \succeq 0, \quad X_{ii} = 1 \;\; \forall i
\]

本例目标：$\max \frac{1}{2}(5 - X_{12} - X_{23} - X_{34} - X_{14} - X_{13})$

\textbf{为什么这是松弛？}原整数解 $x_i\in\{-1,+1\}$ 对应秩-1 的 $X$（$X = xx^T$），而 SDP 允许任意秩的半正定矩阵，可行域更大，最优值 $\geq$ 原整数问题最优值。

\textbf{第三步：从 SDP 解恢复切割（Goemans-Williamson 随机舍入）}

SDP 求解后得到 $X^*$，对其做 Cholesky 分解 $X^* = V^TV$，得到单位向量 $v_1,\ldots,v_4$。

随机舍入：均匀随机选一个超平面法向量 $r \sim \mathcal{N}(0, I)$，令
\[
x_i = \text{sign}(r^T v_i)
\]

\textbf{Goemans-Williamson 定理}：该算法输出的期望切割值 $\geq 0.878 \times \text{OPT}$（$\text{OPT}$ 为真实最优值），这是目前已知最好的多项式时间近似比。

\begin{center}
\begin{tikzpicture}[scale=0.85]
    \draw[->, thick] (-2.5, 0) -- (2.5, 0);
    \draw[->, thick] (0, -2.5) -- (0, 2.5);
    \draw[dashed, orange, thick] (-2, -1.5) -- (2, 1.5) node[right] {随机超平面};

\begin{blackboard}[SDP 例子 2：控制系统稳定性（Lyapunov LMI）]

\textbf{场景}：线性时不变系统 $\dot{x}(t) = Ax(t)$，$x \in \mathbb{R}^n$，判断是否渐近稳定（即 $x(t) \to 0$）。

\textbf{直接方法}：看矩阵 $A$ 的特征值是否全部有负实部——但这不能推广到有不确定性或时变的系统。SDP 方法更通用。

\textbf{Lyapunov 稳定性理论}

\textbf{Lyapunov 函数}：类比"能量"，若能找到一个正的"能量函数" $V(x)$ 沿轨迹严格递减，则系统稳定。

取 $V(x) = x^T P x$，$P \succ 0$（正定二次型）。沿轨迹求导：
\[
\dot{V}(x) = \dot{x}^T P x + x^T P \dot{x}
= x^T A^T P x + x^T P A x
= x^T \underbrace{(A^T P + PA)}_{=:Q} x
\]

要 $V$ 沿轨迹严格递减，需要 $\dot{V}(x) < 0$ 对所有 $x\neq 0$，即 $Q \prec 0$（负定）。

\begin{theorem}[Lyapunov 定理]
线性系统 $\dot{x}=Ax$ 渐近稳定 $\Longleftrightarrow$ 存在 $P \succ 0$ 使得 $A^TP + PA \prec 0$
\end{theorem}

\textbf{建模为 SDP 可行性问题}

变量：对称矩阵 $P \in \mathbb{R}^{n\times n}$（共 $n(n+1)/2$ 个自由度）

\textbf{约束均为线性矩阵不等式（LMI）}：
\[
\text{Find } P \quad \text{s.t.} \quad P \succ 0, \quad A^TP + PA \prec 0
\]

这直接就是 SDP 可行性问题（目标函数取常数 0 即可）。

\textbf{具体例子}：
\[
A = \begin{pmatrix} -1 & 2 \\ 0 & -3 \end{pmatrix}, \qquad
P = \begin{pmatrix} p_{11} & p_{12} \\ p_{12} & p_{22} \end{pmatrix}
\]

展开 $Q = A^TP + PA$：
\[
A^TP + PA
= \begin{pmatrix} -1 & 0 \\ 2 & -3 \end{pmatrix}\begin{pmatrix} p_{11} & p_{12} \\ p_{12} & p_{22} \end{pmatrix}
+ \begin{pmatrix} p_{11} & p_{12} \\ p_{12} & p_{22} \end{pmatrix}\begin{pmatrix} -1 & 2 \\ 0 & -3 \end{pmatrix}
= \begin{pmatrix} -2p_{11} & 2p_{11} - p_{12} - 3p_{12} \\ \cdot & 4p_{12} - 6p_{22} \end{pmatrix}
\]

整理（利用对称性）：
\[
Q = \begin{pmatrix} -2p_{11} & 2p_{11} - 4p_{12} \\ 2p_{11} - 4p_{12} & 4p_{12} - 6p_{22} \end{pmatrix}
\]

\textbf{SDP 求解}：取 $P = I$（$p_{11}=p_{22}=1$，$p_{12}=0$）验证：
\[
Q = \begin{pmatrix} -2 & 2 \\ 2 & -6 \end{pmatrix}
\]

验证 $Q \prec 0$：特征值 $= -4 \pm \sqrt{8} \approx -1.17,\, -6.83$，均 $< 0$。\checkmark

\textbf{进阶：最优化 Lyapunov 函数}

可以把可行性问题改为优化问题，例如最大化 $P$ 最小特征值（增强稳定裕度）：
\[
\max_{P,\,\gamma} \quad \gamma
\quad \text{s.t.} \quad P \succeq \gamma I, \quad A^TP + PA \preceq -\gamma I
\]
仍然是 SDP，但现在有了意义明确的优化目标。

\textbf{推广}：对含不确定性的系统 $\dot{x} = (A + \Delta)x$，$\|\Delta\| \leq \epsilon$，鲁棒稳定性条件也能写成 LMI——这是 SDP 在控制领域的核心优势。

\end{blackboard}

\begin{blackboard}[SDP 例子 3：矩阵补全（核范数最小化）]

\textbf{场景}：观察到矩阵 $M \in \mathbb{R}^{m\times n}$ 的部分元素（由已知下标集 $\Omega$ 索引），恢复完整矩阵。

\textbf{具体例子}：3个用户对4部电影的评分（? 表示未观测）

\[
M = \begin{pmatrix}
5 & ? & 4 & ? \\
? & 3 & ? & 2 \\
4 & ? & 5 & ?
\end{pmatrix}, \quad
\Omega = \{(1,1),(1,3),(2,2),(2,4),(3,1),(3,3)\}
\]

\textbf{第一步：低秩假设}

为什么矩阵是低秩的？用户 $\approx$ 少数几个"口味类型"的线性组合，电影 $\approx$ 少数几个"题材特征"的线性组合，因此评分矩阵 $\approx$ 秩 $r \ll \min(m,n)$ 的矩阵。

秩-1 示例：若所有用户只按"动作 vs 文艺"分类：
\[
M = \underbrace{\begin{pmatrix}0.9\\0.4\\0.8\end{pmatrix}}_{\text{用户偏好}}
\underbrace{\begin{pmatrix}0.9 & 0.6 & 0.8 & 0.5\end{pmatrix}}_{\text{电影特征}}
\]

\textbf{第二步：为何直接最小化秩行不通？}

最直接的建模：$\min_X \text{rank}(X) \;\;\text{s.t.}\;\; X_{ij} = M_{ij}, \forall (i,j)\in\Omega$

但 $\text{rank}(X)$ 是\textbf{非凸函数}（组合优化，NP-Hard）。

\textbf{第三步：核范数——秩的凸松弛}

\textbf{核范数}（nuclear norm）：$\|X\|_* = \sum_k \sigma_k(X)$，其中 $\sigma_k$ 是奇异值。

类比：
\begin{center}
\begin{tabular}{c|c|c}
量 & 非凸原版 & 凸松弛 \\
\hline
向量稀疏度 & $\|x\|_0$（非零元个数） & $\|x\|_1$（L1 范数） \\
矩阵秩 & $\text{rank}(X)$ & $\|X\|_*$（核范数）
\end{tabular}
\end{center}

\textbf{直觉}：$\text{rank}(X)$ = 非零奇异值的个数（L0"范数"），$\|X\|_*$ = 奇异值之和（L1 范数），最小化核范数会促使尽量多的奇异值为零，即低秩。

核范数最小化问题：
\[
\min_X \|X\|_* \quad \text{s.t.} \quad X_{ij} = M_{ij}, \quad \forall (i,j)\in\Omega
\]

\textbf{第四步：转化为 SDP}

\textbf{核范数的 SDP 等价表示}：
\[
\|X\|_* = \min_{W_1, W_2} \;\frac{1}{2}\bigl(\operatorname{tr}(W_1) + \operatorname{tr}(W_2)\bigr)
\quad \text{s.t.} \quad
\begin{pmatrix} W_1 & X \\ X^T & W_2 \end{pmatrix} \succeq 0
\]

\textbf{推导直觉}：由 Schur 补，$\begin{pmatrix}W_1 & X \\ X^T & W_2\end{pmatrix}\succeq 0$ 等价于 $W_1 \succeq XW_2^{-1}X^T$，迫使 $W_1$ 的迹覆盖 $X$ 的"能量"（奇异值），取最小值时恰好等于 $\|X\|_*$。

完整 SDP（变量：$X\in\mathbb{R}^{3\times 4}$，$W_1\in\mathbb{R}^{3\times 3}$，$W_2\in\mathbb{R}^{4\times 4}$）：
\[
\begin{aligned}
\min_{X,W_1,W_2} \quad & \frac{1}{2}\bigl(\operatorname{tr}(W_1) + \operatorname{tr}(W_2)\bigr) \\
\text{s.t.} \quad &
\begin{pmatrix} W_1 & X \\ X^T & W_2 \end{pmatrix} \succeq 0 \\
& X_{11}=5,\; X_{13}=4,\; X_{22}=3,\; X_{24}=2,\; X_{31}=4,\; X_{33}=5
\end{aligned}
\]

\textbf{第五步：解读结果}

SDP 解出后，$X^*$ 即为补全矩阵（满足已知约束的最低秩近似）：

\[
\hat{M} \approx \begin{pmatrix}
5 & 3.0 & 4 & 2.4 \\
3.0 & 3 & 2.4 & 2 \\
4 & 2.4 & 5 & 1.6
\end{pmatrix}
\]

\textbf{理论保证（Candès \& Recht, 2009）}：若真实矩阵秩为 $r$，且观测下标足够随机，则只需观测约 $O(rn\log n)$ 个元素，核范数最小化能以概率趋于 1 精确恢复完整矩阵。

\section{Lagrangian 与对偶函数}
%========================================

\begin{blackboard}[Lagrangian 函数]

考虑优化问题：
\[
\begin{aligned}
\min_{x} \quad & f(x) \\
\text{s.t.} \quad & g_i(x) \leq 0, \quad i = 1, \ldots, m \\
& h_j(x) = 0, \quad j = 1, \ldots, p
\end{aligned}
\]

\begin{definition}[Lagrangian]
\textbf{Lagrangian 函数}：
\[
L(x, \lambda, \nu) = f(x) + \sum_{i=1}^m \lambda_i g_i(x) + \sum_{j=1}^p \nu_j h_j(x)
\]

其中 $\lambda_i \geq 0$ 是不等式约束的\textbf{Lagrange 乘子}，$\nu_j$ 是等式约束的乘子。
\end{definition}

\textbf{向量形式}：$L(x, \lambda, \nu) = f(x) + \lambda^T g(x) + \nu^T h(x)$

\end{blackboard}

\begin{blackboard}[对偶函数]

\begin{definition}[对偶函数]
\textbf{Lagrange 对偶函数}：
\[
g(\lambda, \nu) = \inf_x L(x, \lambda, \nu) = \inf_x \left[ f(x) + \lambda^T g(x) + \nu^T h(x) \right]
\]
\end{definition}

\textbf{关键性质}：对偶函数是\textbf{凹函数}（即使原问题非凸！）

\textbf{原因}：$g$ 是关于 $(\lambda, \nu)$ 的仿射函数的逐点下确界

\end{blackboard}

\begin{blackboard}[对偶函数的下界性质]

\begin{theorem}[弱对偶]
对任意 $\lambda \geq 0$ 和任意 $\nu$：
\[
g(\lambda, \nu) \leq p^*
\]
其中 $p^*$ 是原问题的最优值。
\end{theorem}

\textbf{证明}：设 $\tilde{x}$ 是可行解，则 $g_i(\tilde{x}) \leq 0$，$h_j(\tilde{x}) = 0$。

\begin{align*}
g(\lambda, \nu) &= \inf_x L(x, \lambda, \nu) \\
&\leq L(\tilde{x}, \lambda, \nu) \\
&= f(\tilde{x}) + \underbrace{\lambda^T g(\tilde{x})}_{\leq 0} + \underbrace{\nu^T h(\tilde{x})}_{= 0} \\
&\leq f(\tilde{x})
\end{align*}

对所有可行解都成立，故 $g(\lambda, \nu) \leq p^*$。

\end{blackboard}

\begin{blackboard}[计算例子：拉格朗日乘子法（等式约束）]

\textbf{问题}：求离原点最近的直线 $x_1 + x_2 = 1$ 上的点。
\[
\min_{x_1, x_2} \;\; x_1^2 + x_2^2 \quad \text{s.t.} \quad x_1 + x_2 = 1
\]

\textbf{第一步：写 Lagrangian}（等式约束引入无约束乘子 $\nu \in \mathbb{R}$）
\[
L(x_1, x_2, \nu) = x_1^2 + x_2^2 + \nu(x_1 + x_2 - 1)
\]

\textbf{第二步：一阶最优性条件}（$\nabla_x L = 0$，无不等式约束，直接求导）
\[
\frac{\partial L}{\partial x_1} = 2x_1 + \nu = 0 \implies x_1 = -\frac{\nu}{2}
\]
\[
\frac{\partial L}{\partial x_2} = 2x_2 + \nu = 0 \implies x_2 = -\frac{\nu}{2}
\]

\textbf{第三步：代入等式约束}
\[
x_1 + x_2 = 1 \implies -\frac{\nu}{2} - \frac{\nu}{2} = 1 \implies \nu = -1
\]

\textbf{第四步：代回得最优解}
\[
x_1^* = x_2^* = \frac{1}{2}, \quad \nu^* = -1, \quad f^* = \frac{1}{4} + \frac{1}{4} = \frac{1}{2}
\]

\textbf{几何验证}：直线 $x_1 + x_2 = 1$ 法方向为 $(1,1)/\sqrt{2}$，原点到直线的垂足正是 $(1/2, 1/2)$，距离 $= 1/\sqrt{2}$，$f^* = 1/2 = (1/\sqrt{2})^2$ ✓

\textbf{$\nu$ 的含义（影子价格）}：$\nu^* = -1$ 表示若将约束右端从 1 增大到 $1+\epsilon$，最优值近似变化 $-\nu^* \cdot \epsilon = \epsilon$（距离增大，费用上升）。

\end{blackboard}

\section{弱对偶与强对偶}
%========================================

\begin{blackboard}[对偶问题]

\textbf{原问题}（Primal）：
\[
p^* = \min_x f(x) \quad \text{s.t.} \quad g(x) \leq 0, \; h(x) = 0
\]

\textbf{对偶问题}（Dual）：
\[
d^* = \max_{\lambda \geq 0, \nu} g(\lambda, \nu) = \max_{\lambda \geq 0, \nu} \inf_x L(x, \lambda, \nu)
\]

\textbf{弱对偶}（总是成立）：
\[
d^* \leq p^*
\]

\textbf{对偶间隙}：$p^* - d^*$

\end{blackboard}

\begin{blackboard}[强对偶]

\begin{definition}[强对偶]
如果 $d^* = p^*$，称\textbf{强对偶}成立。
\end{definition}

\textbf{什么时候强对偶成立？}

\begin{theorem}[Slater 条件]
对于凸优化问题，如果存在\textbf{严格可行点} $\tilde{x}$：
\[
g_i(\tilde{x}) < 0, \quad i = 1, \ldots, m; \quad h_j(\tilde{x}) = 0, \quad j = 1, \ldots, p
\]
则强对偶成立。
\end{theorem}

\textbf{直觉}：约束不能``恰好''卡在边界上，要有``余量''

\end{blackboard}

\begin{blackboard}[Slater 条件的证明 $\bigstar$]

\textbf{目标}：证明 $d^* = p^*$。由弱对偶 $d^* \leq p^*$，只需证 $d^* \geq p^*$。

\textbf{证明思路}：构造两个不相交的凸集，用分离超平面定理"找出"对偶最优解，再用 Slater 条件排除退化情形。

\vspace{0.5em}
\textbf{第一步：构造两个不相交的凸集}

定义"可达值域"（含松弛）：
\[
\mathcal{A} = \bigl\{(u,\, v,\, t) \in \mathbb{R}^m \times \mathbb{R}^p \times \mathbb{R}
  \;\big|\; \exists\, x \in \mathcal{D}:\; g(x) \preceq u,\; h(x) = v,\; f(x) \leq t \bigr\}
\]

定义"比最优值更好"的集合：
\[
\mathcal{B} = \{(u, v, t) : u \preceq 0,\; v = 0,\; t < p^*\}
\]

\begin{enumerate}
    \item $\mathcal{A}$ 是凸集：若 $(u_1, v_1, t_1), (u_2, v_2, t_2) \in \mathcal{A}$，由 $f$、$g_i$ 的凸性可验证其凸组合也在 $\mathcal{A}$ 内。
    \item $\mathcal{B}$ 是凸集（线性约束定义的集合）。
    \item $\mathcal{A} \cap \mathcal{B} = \emptyset$：若有 $(u,v,t) \in \mathcal{A} \cap \mathcal{B}$，则存在 $x$ 满足 $g(x) \preceq u \preceq 0$、$h(x) = v = 0$、$f(x) \leq t < p^*$——即 $x$ 可行且 $f(x) < p^*$，与 $p^*$ 是最优值矛盾。
\end{enumerate}

\textbf{第二步：分离超平面定理}

$\mathcal{A}$ 与 $\mathcal{B}$ 均为凸集且不相交，由分离超平面定理，存在 $(\lambda, \nu, \mu) \neq 0$，使得
\[
\lambda^T u + \nu^T v + \mu\, t \;\geq\; \lambda^T s + \mu\, r
\quad \forall\, (u,v,t)\in\mathcal{A},\; (s, 0, r) \in \mathcal{B} \tag{$\star$}
\]

从 $\mathcal{B}$ 的自由度推出符号约束：
\begin{itemize}
    \item 令 $s_i \to -\infty$（$s \preceq 0$ 可任意负）：若某个 $\lambda_i < 0$，右边趋于 $+\infty$，违反 ($\star$)，故 $\lambda \geq 0$。
    \item 令 $r \to -\infty$（$r < p^*$ 可任意负）：同理推出 $\mu \geq 0$。
\end{itemize}

\textbf{第三步：用 Slater 条件证 $\mu > 0$（关键步骤）}

假设 $\mu = 0$。将严格可行点 $\tilde{x}$（满足 $g_i(\tilde{x}) < 0$，$h_j(\tilde{x}) = 0$）代入 ($\star$) 左边，取 $s = 0$，$r$ 任意：

\[
\lambda^T g(\tilde{x}) + \nu^T \underbrace{h(\tilde{x})}_{=0} \;\geq\; \lambda^T \cdot 0 = 0
\;\implies\;
\lambda^T g(\tilde{x}) \geq 0
\]

但 $\lambda \geq 0$ 且 $g_i(\tilde{x}) < 0$，故 $\lambda^T g(\tilde{x}) \leq 0$。

两者合并得 $\lambda^T g(\tilde{x}) = 0$，又因为 $\lambda_i \geq 0$ 且 $g_i(\tilde{x}) < 0$，每个乘积 $\lambda_i g_i(\tilde{x}) \leq 0$，因此每项必须为零，即 $\lambda = 0$。

此时 $\mu = 0$，$\lambda = 0$，但 $(\lambda, \nu, \mu) \neq 0$ 要求 $\nu \neq 0$。然而 $h_j$ 是仿射函数（满射），$\nu^T h(x)$ 无界，无法被 ($\star$) 左边一致下界——矛盾。

故 $\mu > 0$。

\textbf{第四步：构造对偶最优解}

对任意 $x \in \mathcal{D}$，$(u,v,t) = (g(x),h(x),f(x)) \in \mathcal{A}$，在 ($\star$) 中令 $s = 0$，$r \to (p^*)^-$：
\[
\lambda^T g(x) + \nu^T h(x) + \mu f(x) \;\geq\; \mu p^*
\]

两边除以 $\mu > 0$，令 $\lambda^* = \lambda/\mu$，$\nu^* = \nu/\mu$：
\[
f(x) + (\lambda^*)^T g(x) + (\nu^*)^T h(x) \;\geq\; p^* \quad \forall x \in \mathcal{D}
\]

对 $x$ 取下确界，注意左边正是 Lagrangian，得：
\[
g(\lambda^*, \nu^*) = \inf_x\, L(x, \lambda^*, \nu^*) \;\geq\; p^*
\]

故 $d^* = \sup_{\lambda \geq 0,\, \nu} g(\lambda, \nu) \geq g(\lambda^*, \nu^*) \geq p^*$。

\textbf{第五步：结论}

由弱对偶 $d^* \leq p^*$ 与上面的 $d^* \geq p^*$：
\[
\boxed{d^* = p^*} \qquad \square
\]

\end{blackboard}

\begin{insight}[Slater 条件在证明中哪里起了作用？]

证明的关键是排除退化情形 $\mu = 0$：
\begin{itemize}
    \item 若 $\mu = 0$，超平面的法向量中"目标函数方向"消失，超平面完全由约束驱动，无法给出有意义的对偶变量。
    \item Slater 条件（严格可行点）保证了 $\lambda^T g(\tilde{x}) < 0$ 的可能性，迫使 $\lambda = 0$，从而与 $\nu \neq 0$ 产生矛盾。
    \item 本质上：严格可行点使得可行域"有内点"，排除了约束退化（线性无关约束失效）的情形。
\end{itemize}

\textbf{反例}（Boyd \& Vandenberghe）：若 Slater 条件不满足，强对偶可能失败。考虑如下凸优化问题：
\[
\min_{x_1, x_2} \; e^{-x_1} \quad \text{s.t.} \quad \frac{x_1^2}{x_2} \leq 0, \quad \mathcal{D} = \{(x_1, x_2) \mid x_2 > 0\}
\]
目标函数 $e^{-x_1}$ 和约束函数 $x_1^2/x_2$（$x_1^2$ 的透视函数）均为凸函数，这确实是一个凸优化问题。

\textbf{Slater 条件为何失效}：要求存在 $x_2 > 0$ 使 $x_1^2/x_2 < 0$，但分子 $x_1^2 \geq 0$、分母 $x_2 > 0$，故 $x_1^2/x_2 \geq 0$ 恒成立，严格可行点不存在。

\textbf{原问题最优值}：约束 $x_1^2/x_2 \leq 0$ 与定义域 $x_2 > 0$ 联合要求 $x_1 = 0$，可行域为射线 $\{(0, x_2) \mid x_2 > 0\}$，目标值恒为 $e^0 = 1$，故 $p^* = 1$。

\textbf{对偶问题最优值}：拉格朗日函数 $L = e^{-x_1} + \lambda x_1^2/x_2$（$\lambda \geq 0$）。固定任意 $\lambda \geq 0$，令 $x_1 \to +\infty$ 且 $x_2 = x_1^3$，则
\[
e^{-x_1} \to 0, \qquad \lambda \cdot \frac{x_1^2}{x_1^3} = \frac{\lambda}{x_1} \to 0
\]
故 $g(\lambda) = \inf L = 0$，从而 $d^* = \max_{\lambda \geq 0} g(\lambda) = 0$。

\textbf{对偶间隙}：$p^* - d^* = 1 - 0 = 1 > 0$，强对偶失败。

\end{insight}

\begin{blackboard}[具体实例：LP 的 Lagrangian 对偶推导]

上面的理论对 LP 成立，且可以精确写出对偶问题的形式。

\textbf{原始问题}（标准不等式形式）：
\[
p^* = \min_{x \geq 0} \; c^T x \quad \text{s.t.} \quad Ax \geq b
\]

\textbf{第一步：写 Lagrangian}（不等式约束 $b - Ax \leq 0$ 引入 $\lambda \geq 0$，非负约束 $-x \leq 0$ 引入 $\mu \geq 0$）：
\[
L(x, \lambda, \mu)
= c^T x + \lambda^T(b - Ax) - \mu^T x
= \underbrace{(c - A^T\lambda - \mu)^T}_{\text{$x$ 的系数}} x + \lambda^T b
\]

\textbf{第二步：对 $x$ 取下确界}（$x$ 此时无约束，非负已由 $\mu$ 接管）：
\[
g(\lambda, \mu) = \inf_{x \in \mathbb{R}^n} \left[(c - A^T\lambda - \mu)^T x + \lambda^T b\right]
= \begin{cases} \lambda^T b & \text{若 } c - A^T\lambda - \mu = 0 \\ -\infty & \text{否则} \end{cases}
\]

仿射函数 $a^T x$ 若 $a \neq 0$ 则无下界；只有 $a = 0$ 时下确界有限。

\textbf{第三步：写对偶问题}（$\max$ 自动排除 $g = -\infty$，即强制 $\mu = c - A^T\lambda \geq 0$）：
\[
\boxed{d^* = \max_{\lambda \geq 0} \; b^T\lambda \quad \text{s.t.} \quad A^T\lambda \leq c}
\]

\textbf{对称美}：

\begin{center}
\begin{tabular}{c|cc}
 & 原始问题 & 对偶问题 \\
\hline
变量 & $x \geq 0$ & $\lambda \geq 0$ \\
目标 & $\min\; c^T x$ & $\max\; b^T\lambda$ \\
约束 & $Ax \geq b$ & $A^T\lambda \leq c$ \\
\end{tabular}
\end{center}

约束矩阵转置，目标系数与约束右端互换。\textbf{对偶的对偶恢复原始问题}（LP 满足 Slater 条件 $\Leftrightarrow$ 有严格可行的内点，从而强对偶成立，$d^* = p^*$）。

\end{blackboard}

\textbf{1. 提供下界}：即使不能解原问题，对偶给出最优值的界

\textbf{2. 有时对偶更容易}：对偶问题可能有更好的结构

\textbf{3. 最优性条件}：强对偶 + KKT 条件 = 最优性的充要条件

\textbf{4. 敏感性分析}：对偶变量告诉我们约束的``价值''

\end{insight}

\section{KKT 条件}
%========================================

\begin{blackboard}[KKT 条件：最优性的充要条件]

\begin{theorem}[KKT 条件]
对于凸优化问题，如果强对偶成立，则 $x^*$ 是最优解当且仅当存在 $\lambda^*, \nu^*$ 使得：

\textbf{1. 原始可行性}（Primal Feasibility）：
\[
g_i(x^*) \leq 0, \quad h_j(x^*) = 0
\]

\textbf{2. 对偶可行性}（Dual Feasibility）：
\[
\lambda_i^* \geq 0
\]

\textbf{3. 互补松弛}（Complementary Slackness）：
\[
\lambda_i^* g_i(x^*) = 0, \quad \forall i
\]

\textbf{4. 稳定性}（Stationarity）：
\[
\nabla f(x^*) + \sum_i \lambda_i^* \nabla g_i(x^*) + \sum_j \nu_j^* \nabla h_j(x^*) = 0
\]
\end{theorem}

\end{blackboard}

\begin{blackboard}[KKT 条件的推导 $\bigstar$]

\textbf{设强对偶成立}：$p^* = d^*$

\textbf{Step 1}：写出对偶间隙为零的条件

设 $x^*$ 是原问题最优解，$(\lambda^*, \nu^*)$ 是对偶问题最优解

\begin{align*}
0 &= p^* - d^* \\
&= f(x^*) - g(\lambda^*, \nu^*) \\
&= f(x^*) - \inf_x L(x, \lambda^*, \nu^*) \\
&\geq f(x^*) - L(x^*, \lambda^*, \nu^*) \quad \text{（下确界 $\leq$ 任意值）} \\
&= f(x^*) - f(x^*) - \sum_i \lambda_i^* g_i(x^*) - \sum_j \nu_j^* h_j(x^*) \\
&= -\sum_i \underbrace{\lambda_i^*}_{\geq 0} \underbrace{g_i(x^*)}_{\leq 0} - \sum_j \nu_j^* \underbrace{h_j(x^*)}_{= 0} \\
&\geq 0
\end{align*}

\textbf{Step 2}：由 $0 \geq 0$，所有不等式必须取等

\begin{itemize}
    \item $\inf_x L(x, \lambda^*, \nu^*) = L(x^*, \lambda^*, \nu^*)$
    
    $\Rightarrow$ $x^*$ 是 $L(\cdot, \lambda^*, \nu^*)$ 的最小值点
    
    $\Rightarrow$ $\nabla_x L(x^*, \lambda^*, \nu^*) = 0$ \quad \textbf{（稳定性）}
    
    \item $\sum_i \lambda_i^* g_i(x^*) = 0$，且每项 $\lambda_i^* g_i(x^*) \leq 0$
    
    $\Rightarrow$ $\lambda_i^* g_i(x^*) = 0$ 对所有 $i$ \quad \textbf{（互补松弛）}
\end{itemize}

\hfill $\square$

\end{blackboard}

\begin{blackboard}[KKT 条件的直觉]

\textbf{稳定性}：$\nabla_x L(x^*, \lambda^*, \nu^*) = 0$

在最优点，目标函数的梯度被约束的梯度``平衡''

\begin{center}
\begin{tikzpicture}[scale=1.2]
    % 可行域
    \fill[blue!10] (0, 0) -- (3, 0) -- (3, 2) -- (0, 3) -- cycle;
    \draw[blue, thick] (0, 0) -- (3, 0) -- (3, 2) -- (0, 3) -- cycle;
    
    % 等高线
    \draw[gray, dashed] (0.5, 0.5) circle (0.5);
    \draw[gray, dashed] (0.5, 0.5) circle (1);
    \draw[gray, dashed] (0.5, 0.5) circle (1.5);
    
    % 最优点
    \fill[red] (1.5, 1.5) circle (2pt);
    \node[above right] at (1.5, 1.5) {$x^*$};

    \draw[->, thick, red] (1.5, 1.5) -- (0.8, 0.8);

    \draw[->, thick, blue] (1.5, 1.5) -- (2.2, 2.2);
    \node[right, blue] at (2.2, 2.2) {$\lambda^* \nabla g$};
\end{tikzpicture}
\end{center}

\textbf{互补松弛}：$\lambda_i^* g_i(x^*) = 0$

\begin{itemize}
    \item 如果约束不紧（$g_i(x^*) < 0$），则 $\lambda_i^* = 0$（约束``不起作用''）
    \item 如果 $\lambda_i^* > 0$，则约束必须紧（$g_i(x^*) = 0$）
\end{itemize}

\end{blackboard}

\begin{blackboard}[例子：带约束的二次规划]

\textbf{问题}：
\[
\min_x \frac{1}{2}x^2 \quad \text{s.t.} \quad x \geq 1
\]

\textbf{转换}：$g(x) = 1 - x \leq 0$

\textbf{Lagrangian}：$L(x, \lambda) = \frac{1}{2}x^2 + \lambda(1 - x)$

\textbf{KKT 条件}：
\begin{enumerate}
    \item 原始可行：$x \geq 1$
    \item 对偶可行：$\lambda \geq 0$
    \item 互补松弛：$\lambda(1 - x) = 0$
    \item 稳定性：$x - \lambda = 0$
\end{enumerate}

\textbf{求解}：由稳定性 $\lambda = x$。由互补松弛，要么 $\lambda = 0$（则 $x = 0$，违反原始可行），要么 $x = 1$。

故 $x^* = 1$，$\lambda^* = 1$。

\end{blackboard}

\begin{blackboard}[计算例子：KKT 条件（不等式约束，需分情况讨论）]

\textbf{问题}：求平面上满足 $x_1 + 2x_2 \geq 3$ 的区域中离原点最近的点。
\[
\min_{x_1, x_2} \;\; x_1^2 + x_2^2 \quad \text{s.t.} \quad x_1 + 2x_2 \geq 3
\]

\textbf{转换为标准形式}（$g(x) \leq 0$）：令 $g(x) = 3 - x_1 - 2x_2 \leq 0$，乘子 $\lambda \geq 0$。

\textbf{Lagrangian}：
\[
L(x_1, x_2, \lambda) = x_1^2 + x_2^2 + \lambda(3 - x_1 - 2x_2)
\]

\textbf{KKT 条件}：
\begin{enumerate}
    \item \textbf{稳定性}（$\nabla_x L = 0$）：$2x_1 - \lambda = 0$，$2x_2 - 2\lambda = 0$
    \item \textbf{原始可行}：$x_1 + 2x_2 \geq 3$
    \item \textbf{对偶可行}：$\lambda \geq 0$
    \item \textbf{互补松弛}：$\lambda(3 - x_1 - 2x_2) = 0$
\end{enumerate}

\textbf{分两种情况}：

\textbf{情况一：约束不紧（$\lambda = 0$）}

由稳定性：$x_1 = 0, x_2 = 0$。但 $(0,0)$ 不满足 $x_1 + 2x_2 \geq 3$，违反原始可行。\textbf{舍去}。

\textbf{情况二：约束有效（$x_1 + 2x_2 = 3$，$\lambda > 0$）}

由稳定性得 $x_1 = \lambda/2$，$x_2 = \lambda$。代入有效约束：
\[
\frac{\lambda}{2} + 2\lambda = 3 \implies \frac{5\lambda}{2} = 3 \implies \lambda^* = \frac{6}{5}
\]

\[
x_1^* = \frac{3}{5}, \quad x_2^* = \frac{6}{5}, \quad \lambda^* = \frac{6}{5} > 0 \;\checkmark
\]

\textbf{最优值}：$f^* = (3/5)^2 + (6/5)^2 = 9/25 + 36/25 = \dfrac{9}{5}$

\textbf{几何验证}：无约束最优点为原点 $(0,0)$，它在 $x_1+2x_2 < 3$ 一侧（违约），故最优解在约束边界上，且最优点 $\to$ 原点的方向 $(3/5, 6/5) \propto (1, 2)$ 正是约束超平面的法方向 ✓

\textbf{对偶解释}：$\lambda^* = 6/5$ 表示将约束右端从 3 变为 $3+\epsilon$ 时，最优值近似增大 $\lambda^* \epsilon = 6\epsilon/5$（约束越紧，离原点越远）。

\end{blackboard}

\begin{blackboard}[计算例子：等式 + 不等式约束同时存在]

\textbf{问题}：最小化距原点的平方距离，等式约束 $x_1+x_2=2$，同时不能让 $x_1$ 太小。
\[
\min_{x_1, x_2} \;\; x_1^2 + x_2^2 \quad \text{s.t.} \quad x_1 + x_2 = 2, \quad x_1 \geq \frac{3}{2}
\]

\textbf{第一步：先忽略不等式，只用等式}

在 $x_1+x_2=2$ 上，令 $\nabla_x L=0$（只有等式乘子 $\nu$）：
\[
2x_1 + \nu = 0, \quad 2x_2 + \nu = 0 \implies x_1 = x_2 = 1
\]
检查不等式：$x_1 = 1 < 3/2$，\textbf{违反约束}，因此不等式必须被激活。

\textbf{第二步：两个约束同时有效}

联立 $x_1 + x_2 = 2$ 和 $x_1 = 3/2$，立得 $x_2 = 1/2$。

\textbf{Lagrangian}（$\nu$ 无符号要求，$\lambda \geq 0$ 对应 $\frac{3}{2} - x_1 \leq 0$）：
\[
L = x_1^2 + x_2^2 + \nu(x_1 + x_2 - 2) + \lambda\!\left(\tfrac{3}{2} - x_1\right)
\]

\textbf{KKT 稳定性条件}：
\[
2x_1 + \nu - \lambda = 0, \qquad 2x_2 + \nu = 0
\]

代入 $x_2 = 1/2$：$\nu = -2 \times \frac{1}{2} = -1$。

代入 $x_1 = 3/2$：$\lambda = 2x_1 + \nu = 3 + (-1) = 2 > 0$ \checkmark

\textbf{最优解}：
\[
x_1^* = \frac{3}{2},\quad x_2^* = \frac{1}{2},\quad \nu^* = -1,\quad \lambda^* = 2, \quad f^* = \frac{9}{4}+\frac{1}{4} = \frac{10}{4} = \frac{5}{2}
\]

\textbf{与无等式约束对比}：若无 $x_1 \geq 3/2$，最优为 $(1,1)$，$f=2$；加上约束后被推离，$f$ 增大到 $5/2$，增量恰为 $\lambda^* \times (3/2 - 1) = 2 \times 1/2 = 1$。

\end{blackboard}

\begin{blackboard}[计算例子：两个不等式约束，逐一排查哪个有效]

\textbf{问题}：
\[
\min_{x_1, x_2} \;\; (x_1 - 3)^2 + (x_2 - 2)^2 \quad \text{s.t.} \quad \underbrace{x_1 + x_2 \leq 4}_{g_1 \leq 0}, \quad \underbrace{x_2 \leq 1}_{g_2 \leq 0}
\]

\textbf{无约束最优}：$(3, 2)$。检查：$3+2=5>4$（违反 $g_1$），$2>1$（违反 $g_2$）。\textbf{两个约束都破坏}，需要分情况。

\textbf{Lagrangian}（$\lambda_1, \lambda_2 \geq 0$）：
\[
L = (x_1-3)^2 + (x_2-2)^2 + \lambda_1(x_1+x_2-4) + \lambda_2(x_2-1)
\]
稳定性：
\[
2(x_1-3) + \lambda_1 = 0, \qquad 2(x_2-2) + \lambda_1 + \lambda_2 = 0
\]

\textbf{情况 A：仅 $g_1$ 有效（$\lambda_1>0$，$\lambda_2=0$，$x_1+x_2=4$）}

由稳定性：$x_1 = 3 - \lambda_1/2$，$x_2 = 2 - \lambda_1/2$。

代入有效约束：$(3-\lambda_1/2)+(2-\lambda_1/2)=4 \implies \lambda_1 = 1 > 0$ \checkmark

$x_1 = 5/2,\; x_2 = 3/2$。检查 $g_2$：$x_2 = 3/2 > 1$，\textbf{违反 $g_2$}。情况 A 不可行。

\textbf{情况 B：仅 $g_2$ 有效（$\lambda_2>0$，$\lambda_1=0$，$x_2=1$）}

由第一个稳定性方程：$x_1 = 3$（$\lambda_1=0$，$x_1-3$ 的梯度消失）。

由第二个方程：$\lambda_2 = -2(x_2-2) = -2(1-2) = 2 > 0$ \checkmark

检查 $g_1$：$x_1+x_2 = 3+1 = 4 \leq 4$ \checkmark（恰好满足，但 $\lambda_1=0$ 表示不驱动它）

\textbf{情况 C：两个约束同时有效（$\lambda_1, \lambda_2 > 0$）}

$x_2=1, x_1+x_2=4 \implies x_1=3$，与情况 B 一致，但稳定性：$2(3-3)+\lambda_1=0 \implies \lambda_1=0$，退化为情况 B。

\textbf{最优解}：
\[
x_1^* = 3,\quad x_2^* = 1,\quad \lambda_1^* = 0,\quad \lambda_2^* = 2,\quad f^* = 0^2 + 1^2 = 1
\]

\textbf{直觉}：目标点 $(3,2)$ 被 $x_2 \leq 1$ 阻挡，沿 $x_2$ 方向压到 $x_2=1$，而 $x_1$ 方向无约束可以达到最优 $x_1=3$，故只有 $g_2$ 真正起作用。$g_1$ 的约束恰好在边界上，但不影响最优。

\end{blackboard}

\begin{blackboard}[计算例子：非线性约束——球形可行域]

\textbf{问题}：最小化到原点的平方距离，但可行域是以 $(2,2)$ 为中心、半径 $\sqrt{2}$ 的圆盘内部。
\[
\min_{x_1,x_2} \;\; x_1^2 + x_2^2 \quad \text{s.t.} \quad (x_1-2)^2 + (x_2-2)^2 \leq 2
\]

\textbf{无约束最优}：$(0,0)$。检查：$(0-2)^2+(0-2)^2=8 > 2$，违反约束；因此约束有效。

\textbf{Lagrangian}（$g = (x_1-2)^2+(x_2-2)^2-2 \leq 0$，$\lambda \geq 0$）：
\[
L = x_1^2 + x_2^2 + \lambda\bigl[(x_1-2)^2 + (x_2-2)^2 - 2\bigr]
\]

\textbf{KKT 稳定性条件}（注意梯度含非线性项）：
\[
2x_1 + 2\lambda(x_1 - 2) = 0 \implies x_1(1+\lambda) = 2\lambda \implies x_1 = \frac{2\lambda}{1+\lambda}
\]
\[
2x_2 + 2\lambda(x_2 - 2) = 0 \implies x_2 = \frac{2\lambda}{1+\lambda}
\]

由对称性 $x_1 = x_2$，代入有效约束 $(x_1-2)^2+(x_2-2)^2=2$：
\[
2(x_1-2)^2 = 2 \implies x_1 - 2 = \pm 1
\]
因 $\lambda > 0$ 时 $x_1 = \frac{2\lambda}{1+\lambda} \in (0, 2)$，取 $x_1 = 1$，则：
\[
1 = \frac{2\lambda}{1+\lambda} \implies 1 + \lambda = 2\lambda \implies \lambda^* = 1 > 0 \;\checkmark
\]

\textbf{最优解}：
\[
x_1^* = x_2^* = 1, \quad \lambda^* = 1, \quad f^* = 1^2 + 1^2 = 2
\]

\textbf{几何直觉}：原点到圆心 $(2,2)$ 的连线方向为 $(1,1)/\sqrt{2}$，最近的可行点正是沿该方向从圆心出发距离 $\sqrt{2}$ 处，即 $(2,2)-\sqrt{2}\cdot(1,1)/\sqrt{2} = (1,1)$。

\textbf{KKT 的梯度平行条件}：
\[
\underbrace{\nabla f}_{(2,2)} = -\lambda \underbrace{\nabla g}_{(2(x_1-2),\,2(x_2-2))} \Big|_{(1,1)} = -1 \cdot (-2,-2) = (2,2) \;\checkmark
\]
目标函数等高线与约束边界在 $(1,1)$ 处相切，这是非线性约束下 KKT 条件的几何本质。

\end{blackboard}

\begin{blackboard}[计算例子：非线性约束——双曲线可行域]

\textbf{问题}：在圆盘内且满足双曲线约束的区域上，最小化 $x_1+x_2$。
\[
\min_{x_1,x_2} \;\; x_1 + x_2 \quad \text{s.t.} \quad x_1^2 + x_2^2 \leq 9, \quad x_1 x_2 \geq 1
\]

\textbf{直观分析}：目标是最小化 $x_1+x_2$（越小越好），而 $x_1 x_2 \geq 1$ 要求两者之积不能太小。

在 $x_1 x_2 = 1$ 的约束下，由 \textbf{AM-GM 不等式}：
\[
x_1 + x_2 \geq 2\sqrt{x_1 x_2} = 2, \quad \text{等号当且仅当 } x_1 = x_2 = 1
\]
候选点 $(1,1)$：检查 $x_1^2+x_2^2 = 2 \leq 9$ \checkmark。

\textbf{Lagrangian}（$g_1 = x_1^2+x_2^2-9 \leq 0$，$g_2 = 1-x_1x_2 \leq 0$，$\lambda_1,\lambda_2 \geq 0$）：
\[
L = x_1 + x_2 + \lambda_1(x_1^2+x_2^2-9) + \lambda_2(1 - x_1 x_2)
\]

\textbf{KKT 稳定性条件}：
\[
1 + 2\lambda_1 x_1 - \lambda_2 x_2 = 0
\]
\[
1 + 2\lambda_1 x_2 - \lambda_2 x_1 = 0
\]

在 $(1,1)$ 处检查 $g_1$：$1+1-9=-8 < 0$（严格满足），故互补松弛给出 $\lambda_1^* = 0$。

代入两式：
\[
1 + 0 - \lambda_2 \cdot 1 = 0 \implies \lambda_2^* = 1 > 0 \;\checkmark
\]

$g_2$ 的互补松弛：$\lambda_2(1-x_1 x_2)=1\cdot(1-1)=0$ \checkmark

\textbf{最优解}：
\[
x_1^* = x_2^* = 1, \quad \lambda_1^* = 0, \quad \lambda_2^* = 1, \quad f^* = 2
\]

\textbf{梯度平行条件验证}：$\nabla f = (1,1)$，$\nabla g_2 = (-x_2,-x_1)|_{(1,1)} = (-1,-1)$，
\[
(1,1) = -\lambda_2^* \cdot (-1,-1) = 1 \cdot (1,1) \;\checkmark
\]

\textbf{与线性约束对比}：
\begin{itemize}
  \item 线性约束的边界是直线，法向量为常数向量；
  \item 非线性约束（圆、双曲线）的法向量随位置变化，KKT 的"梯度平行"条件自动选出那个法向量恰好与 $\nabla f$ 对齐的点。
\end{itemize}

\end{blackboard}

\section{对偶的经济学解释}
%========================================

\begin{blackboard}[影子价格：约束的边际价值]

考虑带扰动的优化问题：
\[
\begin{aligned}
p^*(\epsilon) = \min_x \quad & f(x) \\
\text{s.t.} \quad & g_i(x) \leq \epsilon_i, \quad i = 1, \ldots, m
\end{aligned}
\]

当 $\epsilon = 0$ 时，就是原问题，最优值为 $p^*(0)$

\textbf{问题}：如果把第 $i$ 个约束放松一点（$\epsilon_i > 0$），最优值会改善多少？

\end{blackboard}

\begin{blackboard}[影子价格定理]

\begin{theorem}[影子价格]
在最优解处，对偶变量 $\lambda_i^*$ 满足：
\[
\lambda_i^* = -\frac{\partial p^*}{\partial \epsilon_i} \bigg|_{\epsilon = 0}
\]
\end{theorem}

\textbf{直觉}：$\lambda_i^*$ 告诉你``放松约束 $i$ 一个单位，目标函数能改善多少''

\textbf{近似公式}：当 $\epsilon$ 很小时
\[
p^*(\epsilon) \approx p^*(0) - \sum_{i=1}^m \lambda_i^* \epsilon_i
\]

\end{blackboard}

\begin{blackboard}[具体例子：生产问题]

\textbf{问题}：最小化成本，受资源约束
\[
\begin{aligned}
\min_x \quad & 2x_1 + 3x_2 \quad \text{（成本）} \\
\text{s.t.} \quad & x_1 + x_2 \geq 10 \quad \text{（产量需求）} \\
& x_1 \leq 8 \quad \text{（机器A产能）} \\
& x_2 \leq 6 \quad \text{（机器B产能）} \\
& x_1, x_2 \geq 0
\end{aligned}
\]

\textbf{转换为标准形式}（$g_i(x) \leq 0$）：
\begin{align*}
g_1(x) &= -x_1 - x_2 + 10 \leq 0 \quad \text{（产量约束）} \\
g_2(x) &= x_1 - 8 \leq 0 \quad \text{（机器A）} \\
g_3(x) &= x_2 - 6 \leq 0 \quad \text{（机器B）}
\end{align*}

\textbf{求解}：$x_1$ 单价更低，应尽量多用机器A。令 $x_1=8$（机器A满产），则产量约束要求 $x_2 \geq 2$，取 $x_2=2$（机器B远未满产）。

\textbf{最优解}：$x_1^* = 8$，$x_2^* = 2$，成本 $= 2(8) + 3(2) = \mathbf{22}$

\textbf{有效约束}：$g_1$（产量恰好满足）、$g_2$（机器A满产）；$g_3$ 不有效（$x_2^*=2 \ll 6$）。

\end{blackboard}

\begin{blackboard}[影子价格的计算与解释]

\textbf{KKT 验证}（$g_1, g_2$ 有效，$\lambda_3=0$）：
\[
L = 2x_1 + 3x_2 + \lambda_1(-x_1-x_2+10) + \lambda_2(x_1-8) + \lambda_3(x_2-6)
\]
稳定性条件：
\[
2 - \lambda_1 + \lambda_2 = 0, \qquad 3 - \lambda_1 + \lambda_3 = 0
\]
代入 $\lambda_3 = 0$：$\lambda_1^* = 3$，$\lambda_2^* = \lambda_1-2 = 1$。两者均 $\geq 0$ \checkmark

\textbf{影子价格}：$\lambda_1^* = 3$，$\lambda_2^* = 1$，$\lambda_3^* = 0$

\medskip
\textbf{约束1}（产量需求 $x_1+x_2 \geq 10$）：$\lambda_1^* = 3$
\begin{itemize}
    \item 若需求降为 9（放松 $\epsilon_1=1$），最优改为 $x_1=8, x_2=1$，成本 $= 19 = 22 - 3$ \checkmark
    \item 预测：$\Delta \text{成本} \approx -\lambda_1^* \epsilon_1 = -3$
\end{itemize}

\textbf{约束2}（机器A产能 $x_1 \leq 8$）：$\lambda_2^* = 1$
\begin{itemize}
    \item $x_2^* = 8$，产能用尽；若产能扩至 9（$\epsilon_2=1$），最优改为 $x_1=9, x_2=1$，成本 $= 21 = 22 - 1$ \checkmark
    \item 每扩容一单位机器A，成本降低 1 元
\end{itemize}

\textbf{约束3}（机器B产能 $x_2 \leq 6$）：$\lambda_3^* = 0$
\begin{itemize}
    \item $x_2^* = 2 \ll 6$，产能大量闲置；扩大机器B产能毫无价值
    \item 互补松弛：$\lambda_3^*(x_2^*-6)=0 \cdot(-4)=0$ \checkmark
\end{itemize}

\end{blackboard}

\begin{blackboard}[验证：放松约束后重新求解]

\textbf{原问题}：需求 $\geq 10$，机器A $\leq 8$
\begin{itemize}
    \item 最优解：$x_1^* = 8$，$x_2^* = 2$，成本 $= 22$
\end{itemize}

\textbf{放松需求}：需求 $\geq 9$（$\epsilon_1 = 1$）
\begin{itemize}
    \item 新最优：$x_1^* = 8$，$x_2^* = 1$，成本 $= 19$
    \item 实际降低：$22 - 19 = 3 = \lambda_1^* \times 1$ \checkmark
\end{itemize}

\textbf{扩大机器A产能}：$x_1 \leq 9$（$\epsilon_2 = 1$）
\begin{itemize}
    \item 新最优：$x_1^* = 9$，$x_2^* = 1$，成本 $= 21$
    \item 实际降低：$22 - 21 = 1 = \lambda_2^* \times 1$ \checkmark
\end{itemize}

\textbf{扩大机器B产能}：$x_2 \leq 7$（$\epsilon_3 = 1$）
\begin{itemize}
    \item 最优仍为 $x_1^* = 8$，$x_2^* = 2$，成本仍 $= 22$
    \item 降低：$0 = \lambda_3^* \times 1$ \checkmark（机器B本就闲置）
\end{itemize}

\textbf{重新审视}：影子价格的符号取决于问题的设定方式！

\end{blackboard}

\begin{blackboard}[影子价格的正确理解]

\textbf{关键公式}：
\[
p^*(\epsilon) \approx p^*(0) - \lambda^{*T} \epsilon
\]

\textbf{对于最小化问题}：
\begin{itemize}
    \item $\epsilon_i > 0$（放松约束）$\Rightarrow$ 可行域变大 $\Rightarrow$ $p^*$ 可能变小
    \item $\lambda_i^* \geq 0$，所以 $-\lambda_i^* \epsilon_i \leq 0$
\end{itemize}

\textbf{经济学直觉}：
\begin{itemize}
    \item $\lambda_i^*$ = 约束 $i$ 的``影子价格''
    \item = 你愿意为``放松约束一个单位''支付的最高价格
    \item = 约束的边际成本
\end{itemize}

\textbf{实际应用}：
\begin{itemize}
    \item 生产计划：应该购买哪种资源？（看哪个 $\lambda_i^*$ 最大）
    \item 电力市场：边际发电成本
    \item 物流优化：运力瓶颈在哪？
\end{itemize}

\end{blackboard}

\begin{blackboard}[互补松弛的经济学意义]

\textbf{互补松弛}：$\lambda_i^* g_i(x^*) = 0$

\textbf{翻译成人话}：

\begin{center}
\begin{tabular}{c|c|c}
情况 & 数学 & 经济学 \\
\hline
资源有剩余 & $g_i(x^*) < 0$ & 供过于求 \\
$\Rightarrow$ 价格为零 & $\lambda_i^* = 0$ & 免费资源无价值 \\
\hline
资源有价值 & $\lambda_i^* > 0$ & 稀缺资源 \\
$\Rightarrow$ 资源用尽 & $g_i(x^*) = 0$ & 供不应求
\end{tabular}
\end{center}

\textbf{直觉}：只有\textbf{稀缺}的资源才有价格！

\end{blackboard}

\begin{insight}[SVM 中的支持向量]

\begin{blackboard}[凸优化思想在深度学习中的应用]

\textbf{1. 分层凸松弛}

神经网络训练非凸，但每一层固定其他层后是凸的：
\begin{itemize}
    \item 固定 $W_2$，优化 $W_1$：凸问题
    \item 交替优化：坐标下降的思想
\end{itemize}

\textbf{2. 正则化 = 凸约束}

$L_2$ 正则化：$\min \text{Loss} + \lambda \|w\|_2^2$

等价于约束优化：$\min \text{Loss}$ s.t. $\|w\|_2 \leq R$

对偶变量 $\lambda$ = 约束的影子价格！

\textbf{3. 对偶在 Kernel 方法中的核心作用}

SVM 对偶：
\[
\max_\alpha \sum_i \alpha_i - \frac{1}{2} \sum_{i,j} \alpha_i \alpha_j y_i y_j K(x_i, x_j)
\]

核技巧（Kernel Trick）完全依赖对偶形式！

\textbf{4. KKT 条件 $\to$ 最优性检验}

即使用 SGD 训练，也可以用 KKT 残差判断是否收敛

\end{blackboard}

\begin{blackboard}[Logistic 回归的凸性证明 $\bigstar$]

\textbf{损失函数}：
\[
L(w) = \sum_{i=1}^n \log(1 + e^{-y_i w^T x_i})
\]

\textbf{要证}：$L(w)$ 是凸函数

\textbf{证明}：设 $\ell(z) = \log(1 + e^{-z})$，则 $L(w) = \sum_i \ell(y_i w^T x_i)$

\textbf{Step 1}：证明 $\ell(z)$ 是凸的
\[
\ell'(z) = \frac{-e^{-z}}{1 + e^{-z}} = \frac{-1}{1 + e^z} = -(1 - \sigma(z))
\]
\[
\ell''(z) = \sigma(z)(1 - \sigma(z)) \geq 0 \quad \checkmark
\]

\textbf{Step 2}：$\ell(y_i w^T x_i)$ 是 $w$ 的凸函数

因为 $y_i w^T x_i$ 是 $w$ 的仿射函数，凸函数复合仿射仍是凸

\textbf{Step 3}：凸函数的和是凸函数 \hfill $\square$

\textbf{加上正则化}：$L(w) + \lambda \|w\|_2^2$ 是\textbf{强凸}的（$\lambda > 0$）

\end{blackboard}

\begin{blackboard}[什么是马尔可夫性？]

\textbf{马尔可夫性}（Markov Property）：未来只依赖于现在，与过去无关

\[
P(s_{t+1} | s_t, s_{t-1}, \ldots, s_0) = P(s_{t+1} | s_t)
\]

\textbf{例子：天气}

\begin{center}
\begin{tikzpicture}[scale=0.9, 
    state/.style={circle, draw, minimum size=1.2cm},
    every edge/.style={draw, ->, >=stealth, thick}]
    
    \node[state, fill=yellow!30] (sunny) at (0, 0) {晴};
    \node[state, fill=gray!30] (rainy) at (4, 0) {雨};
    
    \draw (sunny) edge[loop above] node[above] {0.8} (sunny);
    \draw (sunny) edge[bend left] node[above] {0.2} (rainy);
    \draw (rainy) edge[bend left] node[below] {0.4} (sunny);
    \draw (rainy) edge[loop above] node[above] {0.6} (rainy);
\end{tikzpicture}
\end{center}

明天是否下雨，只取决于今天的天气，不取决于昨天、前天...

\textbf{马尔可夫链}：状态按概率转移，没有决策

\textbf{马尔可夫决策过程 (MDP)}：加入决策！你可以选择动作影响转移

\end{blackboard}

\begin{blackboard}[MDP 的正式定义]

\textbf{MDP} = $(S, A, P, R, \gamma)$

\begin{itemize}
    \item $S$：状态空间（有限集合 $\{1, 2, \ldots, n\}$）
    \item $A$：动作空间（有限集合）
    \item $P(s' | s, a)$：转移概率（在状态 $s$ 执行动作 $a$ 后到达 $s'$ 的概率）
    \item $R(s, a)$：即时奖励（在状态 $s$ 执行动作 $a$ 获得的奖励）
    \item $\gamma \in [0, 1)$：折扣因子（未来奖励的衰减）
\end{itemize}

\textbf{策略}：$\pi: S \to A$，告诉你在每个状态该做什么

\textbf{目标}：找最优策略 $\pi^*$，最大化累积折扣奖励
\[
\mathbb{E}\left[ \sum_{t=0}^\infty \gamma^t R(s_t, a_t) \right]
\]

\textbf{为什么要折扣？}
\begin{itemize}
    \item 今天的 1 元比明天的 1 元更值钱
    \item 数学上保证级数收敛
    \item $\gamma$ 接近 1：更有远见；$\gamma$ 接近 0：更短视
\end{itemize}

\end{blackboard}

\subsection{Bellman 方程}

\begin{blackboard}[值函数：衡量状态的好坏]

\begin{definition}[状态值函数]
策略 $\pi$ 下，从状态 $s$ 出发的期望累积奖励：
\[
V^\pi(s) = \mathbb{E}_\pi \left[ \sum_{t=0}^\infty \gamma^t R(s_t, \pi(s_t)) \mid s_0 = s \right]
\]
\end{definition}

\textbf{直觉}：$V^\pi(s)$ 告诉你``从状态 $s$ 出发，按策略 $\pi$ 走，期望能拿多少奖励''

\textbf{最优值函数}：
\[
V^*(s) = \max_\pi V^\pi(s)
\]

这是所有策略中能达到的最大值

\end{blackboard}

\begin{blackboard}[Bellman 方程：递归结构]

\begin{theorem}[Bellman 期望方程]
\[
V^\pi(s) = R(s, \pi(s)) + \gamma \sum_{s'} P(s' | s, \pi(s)) V^\pi(s')
\]
\end{theorem}

\textbf{翻译}：从 $s$ 出发的总价值 = 立即奖励 + 折扣 $\times$ 下一状态的期望价值

\begin{theorem}[Bellman 最优方程]
\[
V^*(s) = \max_a \left[ R(s, a) + \gamma \sum_{s'} P(s' | s, a) V^*(s') \right]
\]
\end{theorem}

\textbf{翻译}：最优价值 = 选择最好的动作，获得立即奖励 + 折扣 $\times$ 下一状态的最优价值

\end{blackboard}

\begin{blackboard}[机器维护：计算值函数]

设 $\gamma = 0.9$，Bellman 最优方程：

\textbf{状态``好''}：
\[
V^*(\text{好}) = \max \begin{cases}
10 + 0.9 (0.7 V^*(\text{好}) + 0.3 V^*(\text{中})) & \text{不维护} \\
7 + 0.9 \cdot V^*(\text{好}) & \text{维护}
\end{cases}
\]

\textbf{状态``中''}：
\[
V^*(\text{中}) = \max \begin{cases}
5 + 0.9 (0.6 V^*(\text{中}) + 0.4 V^*(\text{差})) & \text{不维护} \\
2 + 0.9 (0.5 V^*(\text{好}) + 0.5 V^*(\text{中})) & \text{维护}
\end{cases}
\]

\textbf{状态``差''}：
\[
V^*(\text{差}) = \max \begin{cases}
-1 + 0.9 \cdot V^*(\text{差}) & \text{不维护} \\
-4 + 0.9 (0.2 V^*(\text{好}) + 0.6 V^*(\text{中}) + 0.2 V^*(\text{差})) & \text{维护}
\end{cases}
\]

这是一个非线性方程组（因为有 max）... 如何求解？

\end{blackboard}

\subsection{MDP 的线性规划形式}

\begin{blackboard}[关键观察：把 max 变成不等式]

Bellman 最优方程：
\[
V^*(s) = \max_a \left[ R(s, a) + \gamma \sum_{s'} P(s'|s,a) V^*(s') \right]
\]

\textbf{等价于}：$V^*(s)$ 是满足以下条件的\textbf{最小}的 $V$

对所有状态 $s$ 和动作 $a$：
\[
V(s) \geq R(s, a) + \gamma \sum_{s'} P(s'|s,a) V(s')
\]

\textbf{为什么？}
\begin{itemize}
    \item $V^*$ 显然满足这些不等式（因为 max $\geq$ 任何一项）
    \item 如果 $V$ 满足所有不等式且 $V < V^*$，则存在某个 $s$ 使得
    
    $V(s) < V^*(s) = R(s, a^*) + \gamma \sum_{s'} P(s'|s,a^*) V^*(s')$
    
    但 $V(s) \geq R(s, a^*) + \gamma \sum_{s'} P(s'|s,a^*) V(s') \geq R(s, a^*) + \gamma \sum_{s'} P(s'|s,a^*) V^*(s')$
    
    矛盾！
\end{itemize}

\end{blackboard}

\begin{blackboard}[MDP 的 LP 形式]

\begin{theorem}[MDP 的线性规划]
最优值函数 $V^*$ 是以下 LP 的解：
\[
\begin{aligned}
\min_V \quad & \sum_s \mu(s) V(s) \\
\text{s.t.} \quad & V(s) \geq R(s,a) + \gamma \sum_{s'} P(s'|s,a) V(s'), \quad \forall s \in S, \forall a \in A
\end{aligned}
\]
其中 $\mu(s) > 0$ 是任意正的权重。
\end{theorem}

\textbf{这是一个标准的 LP！}
\begin{itemize}
    \item 变量：$V(s)$，共 $|S|$ 个
    \item 约束：每对 $(s, a)$ 一个，共 $|S| \times |A|$ 个
    \item 目标：线性
    \item 约束：线性
\end{itemize}

\textbf{可以用单纯形法或内点法求解！}

\end{blackboard}

\begin{blackboard}[机器维护问题的 LP]

状态：好(G)、中(M)、差(B)，动作：不维护(N)、维护(R)

\textbf{变量}：$V_G, V_M, V_B$

\textbf{LP}（取 $\mu = (1/3, 1/3, 1/3)$）：
\[
\min \frac{1}{3}(V_G + V_M + V_B)
\]

\textbf{约束}（6 个，每个 $(s,a)$ 一个）：
\begin{align*}
V_G &\geq 10 + 0.9(0.7 V_G + 0.3 V_M) & \text{(好, 不维护)} \\
V_G &\geq 7 + 0.9 V_G & \text{(好, 维护)} \\
V_M &\geq 5 + 0.9(0.6 V_M + 0.4 V_B) & \text{(中, 不维护)} \\
V_M &\geq 2 + 0.9(0.5 V_G + 0.5 V_M) & \text{(中, 维护)} \\
V_B &\geq -1 + 0.9 V_B & \text{(差, 不维护)} \\
V_B &\geq -4 + 0.9(0.2 V_G + 0.6 V_M + 0.2 V_B) & \text{(差, 维护)}
\end{align*}

\end{blackboard}

\begin{blackboard}[用 CVXPY 求解]

\begin{lstlisting}[language=Python]
import cvxpy as cp
import numpy as np

# 变量
V = cp.Variable(3)  # V[0]=V_G, V[1]=V_M, V[2]=V_B
gamma = 0.9

# 目标
obj = cp.Minimize(cp.sum(V) / 3)

# 约束
constraints = [
    # 状态 好
    V[0] >= 10 + gamma * (0.7*V[0] + 0.3*V[1]),  # 不维护
    V[0] >= 7 + gamma * V[0],                     # 维护
    # 状态 中
    V[1] >= 5 + gamma * (0.6*V[1] + 0.4*V[2]),   # 不维护
    V[1] >= 2 + gamma * (0.5*V[0] + 0.5*V[1]),   # 维护
    # 状态 差
    V[2] >= -1 + gamma * V[2],                    # 不维护
    V[2] >= -4 + gamma * (0.2*V[0] + 0.6*V[1] + 0.2*V[2])  # 维护
]

prob = cp.Problem(obj, constraints)
prob.solve()
print(f"V* = {V.value}")  # [73.8, 64.5, 47.2] 大约
\end{lstlisting}

\end{blackboard}

\subsection{LP 对偶与占用度量}

\begin{blackboard}[MDP 的对偶 LP]

\textbf{原问题}有对偶！引入对偶变量 $\rho(s, a) \geq 0$

\begin{theorem}[MDP 的对偶 LP]
\[
\begin{aligned}
\max_\rho \quad & \sum_{s,a} \rho(s,a) R(s,a) \\
\text{s.t.} \quad & \sum_a \rho(s,a) = \mu(s) + \gamma \sum_{s',a'} \rho(s',a') P(s|s',a'), \quad \forall s \\
& \rho(s,a) \geq 0
\end{aligned}
\]
\end{theorem}

\textbf{$\rho(s,a)$ 的含义}：\textbf{占用度量}（Occupancy Measure）

$\rho(s,a)$ = 在最优策略下，处于状态 $s$ 并执行动作 $a$ 的（折扣）频率

\end{blackboard}

\begin{blackboard}[对偶约束的直觉：流量守恒]

约束：$\sum_a \rho(s,a) = \mu(s) + \gamma \sum_{s',a'} \rho(s',a') P(s|s',a')$

\textbf{翻译}：
\[
\underbrace{\sum_a \rho(s,a)}_{\text{从 } s \text{ 流出}} = \underbrace{\mu(s)}_{\text{初始流入}} + \underbrace{\gamma \sum_{s',a'} \rho(s',a') P(s|s',a')}_{\text{从其他状态转移来}}
\]

\begin{center}
\begin{tikzpicture}[scale=0.9]
    \node[draw, circle, fill=blue!20, minimum size=1.5cm] (s) at (0, 0) {$s$};
    
    % 流入
    \draw[->, thick, red] (-3, 1) -- (s) node[midway, above] {$\mu(s)$};
    \draw[->, thick, orange] (-3, -0.5) -- (s) node[midway, below] {从 $s'$ 来};
    
    % 流出
    \draw[->, thick, blue] (s) -- (3, 0.5) node[midway, above] {$\rho(s, a_1)$};
    \draw[->, thick, blue] (s) -- (3, -0.5) node[midway, below] {$\rho(s, a_2)$};
    
    \node at (0, -2) {流入 = 流出};
\end{tikzpicture}
\end{center}

\textbf{这就是网络流的守恒定律！}

\end{blackboard}

\begin{blackboard}[原问题 vs 对偶问题]

\begin{center}
\begin{tabular}{c|c|c}
& \textbf{原问题} & \textbf{对偶问题} \\
\hline
\textbf{变量} & $V(s)$：值函数 & $\rho(s,a)$：占用度量 \\
\textbf{数量} & $|S|$ 个 & $|S| \times |A|$ 个 \\
\textbf{约束} & Bellman 不等式 & 流量守恒 \\
\textbf{目标} & 最小化总值 & 最大化总奖励 \\
\textbf{视角} & ``状态有多好？'' & ``应该怎么做？''
\end{tabular}
\end{center}

\textbf{强对偶}：两个 LP 的最优值相等！（凸优化的对偶理论）

\textbf{从对偶解恢复策略}：
\[
\pi^*(a|s) = \frac{\rho^*(s,a)}{\sum_{a'} \rho^*(s,a')}
\]

\end{blackboard}

\begin{blackboard}[互补松弛在 MDP 中的意义]

\textbf{KKT 互补松弛}：$\rho^*(s,a) \cdot \left[ V^*(s) - R(s,a) - \gamma \sum_{s'} P(s'|s,a) V^*(s') \right] = 0$

\textbf{翻译}：
\begin{itemize}
    \item 如果 $\rho^*(s,a) > 0$（动作 $a$ 在状态 $s$ 被执行）
    
    $\Rightarrow$ Bellman 不等式取等（$a$ 是最优动作）
    
    \item 如果动作 $a$ 不是最优的（Bellman 不等式严格）
    
    $\Rightarrow$ $\rho^*(s,a) = 0$（这个动作不会被执行）
\end{itemize}

\textbf{经济学解释}：
\begin{itemize}
    \item $V^*(s)$ = 状态 $s$ 的``影子价格''
    \item 只有``物有所值''的动作才会被使用
    \item 次优动作的``使用量''为零
\end{itemize}

\end{blackboard}

\begin{insight}[MDP-LP 总结]

\textbf{核心结论}：
\begin{itemize}
    \item MDP 的最优策略可以通过 LP 精确求解
    \item 原问题变量是值函数，对偶变量是占用度量
    \item LP 对偶揭示了``评估''与``决策''的深层联系
\end{itemize}

\textbf{为什么这很重要？}
\begin{itemize}
    \item 把决策问题转化为我们熟悉的凸优化
    \item KKT 条件给出最优性的刻画
    \item 影子价格、互补松弛都有自然解释
\end{itemize}

\textbf{局限性}：
\begin{itemize}
    \item 需要知道完整的 MDP 模型（$P$, $R$）
    \item 状态空间大时（如围棋 $10^{170}$ 状态），LP 不可行
    \item 实际中常用近似方法（值迭代、策略迭代、函数逼近）
\end{itemize}

\textbf{但 LP 视角提供了理论基础！}

\end{insight}

\section{GPU 改变现代优化}

\begin{blackboard}[经典优化 vs 现代优化]

\textbf{经典优化}（1960s-2000s）：

\begin{itemize}
    \item 内点法、单纯形法
    \item 每步计算量大，但步数少
    \item 适合中等规模问题（$n < 10^5$）
    \item 需要精确的二阶信息（Hessian）
\end{itemize}

\textbf{现代优化}（2010s-）：

\begin{itemize}
    \item 随机梯度下降（SGD）及其变体
    \item 每步计算量小，但步数多
    \item 适合超大规模问题（$n > 10^9$）
    \item 只需要一阶信息（梯度）
\end{itemize}

\textbf{为什么变了？}

\end{blackboard}

\begin{gpubox}[GPU 的出现改变了游戏规则]

\textbf{内点法的瓶颈}：

每步需要解线性系统 $H \Delta x = -g$，其中 $H$ 是 Hessian

\begin{itemize}
    \item 形成 $H$：$O(n^2)$ 存储
    \item 求解线性系统：$O(n^3)$ 或稀疏情况 $O(n^{1.5})$
    \item \textbf{很难并行化}
\end{itemize}

\textbf{梯度下降的优势}：

每步只需计算 $x \leftarrow x - \alpha \nabla f(x)$

\begin{itemize}
    \item 只需 $O(n)$ 存储
    \item 计算梯度：可以高度并行化
    \item \textbf{完美匹配 GPU 架构}
\end{itemize}

\end{gpubox}

\begin{blackboard}[数值对比]

\textbf{问题}：$n = 10^6$ 变量的凸优化

\begin{center}
\begin{tabular}{c|c|c|c}
方法 & 每步复杂度 & 收敛步数 & 实际时间 \\
\hline
内点法 (CPU) & $O(n^3)$ & $\sim 50$ & 几天（内存不够） \\
梯度下降 (CPU) & $O(n)$ & $\sim 10^4$ & 几小时 \\
梯度下降 (GPU) & $O(n)$ 并行 & $\sim 10^4$ & \textbf{几分钟}
\end{tabular}
\end{center}

\textbf{深度学习的选择}：

神经网络参数量：$n \sim 10^9$（GPT-3 有 1750 亿参数）

内点法？绝对不可能。只能用 SGD + GPU。

\end{blackboard}

\begin{blackboard}[批量优化：GPU 的超能力]

\textbf{场景}：需要解 1000 个独立的小优化问题

\textbf{传统方法}（顺序）：
\begin{itemize}
    \item 每个问题 0.1 秒
    \item 总时间：$1000 \times 0.1 = 100$ 秒
\end{itemize}

\textbf{GPU 方法}（并行）：
\begin{itemize}
    \item 1000 个问题同时求解
    \item 总时间：$\approx 0.1$ 秒（几乎和 1 个问题一样）
\end{itemize}

\textbf{加速比}：$\sim 1000\times$

\textbf{应用}：
\begin{itemize}
    \item 强化学习：同时模拟 1000 个环境
    \item 超参数搜索：同时训练 1000 个模型
    \item 蒙特卡洛：同时采样 1000 条路径
\end{itemize}

\end{blackboard}

\begin{insight}[理论 vs 实践的张力]

\textbf{理论}说：

凸优化用内点法，$O(\sqrt{n})$ 步收敛，每步 $O(n^3)$

\textbf{实践}说：

管他收敛多少步，能在 GPU 上跑、能处理 $10^9$ 变量才是王道

\textbf{调和}：

\begin{itemize}
    \item 小问题（$n < 10^4$）：内点法仍是王者
    \item 大问题（$n > 10^6$）：一阶方法 + GPU
    \item 结构化问题：针对性算法（如 ADMM）
\end{itemize}

\textbf{本课程}：两种视角都要学，知道何时用哪种

\end{insight}

\subsection{实验 2a：CVXPY 求解 SVM 对偶}

\begin{experiment}[观察 KKT 条件与支持向量]

\textbf{目标}：
\begin{itemize}
    \item 用 CVXPY 求解 SVM 对偶问题
    \item 观察互补松弛条件
    \item 识别支持向量
\end{itemize}

\textbf{SVM 原始问题}：
\[
\min_{w, b} \frac{1}{2}\|w\|^2 \quad \text{s.t.} \quad y_i(w^T x_i + b) \geq 1
\]

\textbf{SVM 对偶问题}：
\[
\max_{\alpha} \sum_i \alpha_i - \frac{1}{2} \sum_{i,j} \alpha_i \alpha_j y_i y_j x_i^T x_j \quad \text{s.t.} \quad \alpha_i \geq 0, \sum_i \alpha_i y_i = 0
\]

\end{experiment}

\begin{blackboard}[实验代码框架]

\begin{lstlisting}[language=Python]
import cvxpy as cp
import numpy as np

# 生成数据
np.random.seed(42)
n_samples = 50
X_pos = np.random.randn(n_samples//2, 2) + [2, 2]
X_neg = np.random.randn(n_samples//2, 2) + [-2, -2]
X = np.vstack([X_pos, X_neg])
y = np.array([1]*(n_samples//2) + [-1]*(n_samples//2))

# 对偶问题
n = len(y)
alpha = cp.Variable(n)
K = (X @ X.T) * np.outer(y, y)  # Gram 矩阵

objective = cp.Maximize(cp.sum(alpha) - 0.5 * cp.quad_form(alpha, K))
constraints = [alpha >= 0, alpha @ y == 0]
prob = cp.Problem(objective, constraints)
prob.solve()

# 找支持向量
alpha_val = alpha.value
support_vectors = np.where(alpha_val > 1e-5)[0]
print(f"支持向量数量: {len(support_vectors)}")
print(f"支持向量的 alpha: {alpha_val[support_vectors]}")
\end{lstlisting}

\end{blackboard}

\begin{blackboard}[观察互补松弛]

\textbf{KKT 互补松弛}：$\alpha_i (1 - y_i(w^T x_i + b)) = 0$

\begin{lstlisting}[language=Python]
# 从对偶解恢复原始变量
w = sum(alpha_val[i] * y[i] * X[i] for i in range(n))
# 从支持向量计算 b
sv_idx = support_vectors[0]
b = y[sv_idx] - w @ X[sv_idx]

# 验证互补松弛
for i in range(n):
    margin = y[i] * (w @ X[i] + b)
    slack = alpha_val[i] * (1 - margin)
    if alpha_val[i] > 1e-5:
        print(f"点 {i}: alpha={alpha_val[i]:.4f}, "
              f"margin={margin:.4f}, slack={slack:.6f}")
\end{lstlisting}

\textbf{预期结果}：
\begin{itemize}
    \item $\alpha_i > 0$ 的点：margin $\approx 1$（在边界上）
    \item $\alpha_i = 0$ 的点：margin $> 1$（远离边界）
\end{itemize}

\end{blackboard}

\begin{blackboard}[实验代码：小规模 QP]

\begin{lstlisting}[language=Python]
import cvxpy as cp
import numpy as np
import time

# 小规模 QP: n = 100
n = 100
np.random.seed(42)
P = np.random.randn(n, n)
P = P.T @ P + 0.1 * np.eye(n)  # 确保正定
q = np.random.randn(n)

x = cp.Variable(n)
objective = cp.Minimize(0.5 * cp.quad_form(x, P) + q @ x)
constraints = [x >= 0]
prob = cp.Problem(objective, constraints)

start = time.time()
prob.solve()
print(f"CVXPY (n={n}): {time.time() - start:.4f} 秒")

\section{附录：求解 MDP 的三条路 $\bigstar$}
%========================================

\begin{blackboard}[三条路的出发点]

MDP 的核心目标是找最优策略 $\pi^*$，等价于求最优值函数 $V^*$ 满足 Bellman 最优方程：
\[
V^*(s) = \max_a \left[ r(s,a) + \gamma \sum_{s'} P(s'|s,a)\, V^*(s') \right]
\]

这是一个关于 $V^* \in \mathbb{R}^{|S|}$ 的\textbf{非线性不动点方程}（非线性来自 $\max$）。三条路各有不同的解法思路：

\begin{center}
\begin{tabular}{c|c|c|c}
 & 值迭代 & 策略迭代 & 强化学习 \\
\hline
需要知道 $P$？ & 是 & 是 & 否 \\
每步计算量 & 轻 & 重（解方程组） & 轻（采样） \\
收敛速度 & 线性（$\gamma$ 压缩） & 有限步必收敛 & 慢（方差大） \\
适合规模 & 小规模 & 小规模 & 大规模
\end{tabular}
\end{center}

\end{blackboard}

\begin{blackboard}[第一条路：值迭代（Value Iteration）]

\textbf{核心思路}：直接对 Bellman 最优算子 $T$ 反复迭代。

定义 Bellman 最优算子 $T: \mathbb{R}^{|S|} \to \mathbb{R}^{|S|}$：
\[
(TV)(s) = \max_a \left[ r(s,a) + \gamma \sum_{s'} P(s'|s,a)\, V(s') \right]
\]

\textbf{算法}：从任意初始值 $V^{(0)}$（如全零）出发，迭代：
\[
V^{(t+1)}(s) = \max_a \left[ r(s,a) + \gamma \sum_{s'} P(s'|s,a)\, V^{(t)}(s') \right]
\]

\textbf{收敛性证明}：$T$ 是 $\gamma$-压缩映射（在 $\|\cdot\|_\infty$ 意义下）。

\begin{lemma}[$T$ 是 $\gamma$-压缩映射]
对任意 $V, V' \in \mathbb{R}^{|S|}$：
\[
\|TV - TV'\|_\infty \leq \gamma \|V - V'\|_\infty
\]
\end{lemma}

\textbf{证明}：对任意状态 $s$，设 $a^* = \arg\max_a [r(s,a) + \gamma \sum_{s'} P(s'|s,a) V(s')]$，
\begin{align*}
(TV)(s) - (TV')(s)
&\leq r(s,a^*) + \gamma \sum_{s'} P(s'|s,a^*) V(s') \\
&\quad - r(s,a^*) - \gamma \sum_{s'} P(s'|s,a^*) V'(s') \\
&= \gamma \sum_{s'} P(s'|s,a^*) \bigl[V(s') - V'(s')\bigr] \\
&\leq \gamma \sum_{s'} P(s'|s,a^*) \|V - V'\|_\infty \\
&= \gamma \|V - V'\|_\infty
\end{align*}

对称地，$(TV')(s) - (TV)(s) \leq \gamma\|V-V'\|_\infty$，故 $|(TV)(s)-(TV')(s)| \leq \gamma\|V-V'\|_\infty$，对所有 $s$ 取上确界得结论。$\square$

由 Banach 不动点定理，$T$ 有唯一不动点 $V^* = TV^*$，且从任意初始点出发：
\[
\|V^{(t)} - V^*\|_\infty \leq \gamma^t \|V^{(0)} - V^*\|_\infty
\]
误差每步缩小 $\gamma$ 倍，\textbf{线性收敛}，步数 $O\!\left(\frac{\log(1/\varepsilon)}{1-\gamma}\right)$。

\textbf{每步代价}：遍历所有 $(s,a)$ 对，计算量 $O(|S|^2 |A|)$（含转移概率求和）。

\textbf{缺点}：状态空间爆炸（Atari 游戏的像素空间 $\sim 10^{100000}$）完全无法枚举。

\end{blackboard}

\begin{blackboard}[第二条路：策略迭代（Policy Iteration）]

\textbf{核心思路}：交替做「策略评估」和「策略改进」。

\textbf{算法框架}：从任意初始策略 $\pi^{(0)}$ 出发，交替执行：

\textbf{步骤 1——策略评估}（Policy Evaluation）：固定策略 $\pi$，求对应的值函数 $V^\pi$。

$V^\pi$ 满足线性 Bellman 方程：
\[
V^\pi(s) = r(s, \pi(s)) + \gamma \sum_{s'} P(s' | s, \pi(s))\, V^\pi(s')
\]

写成矩阵形式：设 $[P_\pi]_{ss'} = P(s'|s,\pi(s))$，$r_\pi(s) = r(s,\pi(s))$，
\[
V^\pi = r_\pi + \gamma P_\pi V^\pi
\implies
\underbrace{(I - \gamma P_\pi)}_{\text{可逆，因 $\gamma < 1$}} V^\pi = r_\pi
\implies
V^\pi = (I - \gamma P_\pi)^{-1} r_\pi
\]

$(I - \gamma P_\pi)$ 可逆的原因：$\gamma < 1$ 使其严格对角占优。可以直接求逆 $O(|S|^3)$ 或用迭代法（Gauss-Seidel）。

\textbf{步骤 2——策略改进}（Policy Improvement）：用 $V^\pi$ 贪心更新策略：
\[
\pi'(s) = \arg\max_a \left[ r(s,a) + \gamma \sum_{s'} P(s'|s,a)\, V^\pi(s') \right]
\]

\begin{theorem}[策略改进定理]
$V^{\pi'} \geq V^\pi$（逐分量不等式），且若 $\pi' = \pi$，则 $\pi$ 已是最优策略。
\end{theorem}

\textbf{证明}（单步不等式）：

\begin{align*}
V^\pi(s) &= r(s,\pi(s)) + \gamma \sum_{s'} P(s'|s,\pi(s)) V^\pi(s') \\
&\leq \max_a \left[r(s,a) + \gamma \sum_{s'} P(s'|s,a) V^\pi(s')\right] \\
&= r(s,\pi'(s)) + \gamma \sum_{s'} P(s'|s,\pi'(s)) V^\pi(s') \\
&\leq r(s,\pi'(s)) + \gamma \sum_{s'} P(s'|s,\pi'(s)) V^{\pi'}(s') \\
&= V^{\pi'}(s) \qquad \square
\end{align*}
（第二个不等号递归展开，需用 $V^\pi \leq V^{\pi'}$ 归纳。）

\textbf{收敛性}：策略空间有限（最多 $|A|^{|S|}$ 个确定性策略），每次迭代严格改进（$V^{\pi'} > V^\pi$ 除非已最优），故\textbf{有限步内必收敛}，与 $\gamma$ 无关。

\textbf{与值迭代的比较}：策略迭代每步计算量大（$O(|S|^3)$ 解线性方程组），但步数通常远少于值迭代——实践中几步到几十步即可收敛。

\end{blackboard}

\begin{blackboard}[第三条路：强化学习（不知道 $P$）]

前两条路都假设已知转移概率 $P(s'|s,a)$（model-based）。现实中 $P$ 往往未知，只能通过与环境交互采样得到 $(s, a, r, s')$。

\textbf{Q-learning}（值函数法，off-policy）

维护动作值函数 $Q(s,a)$ 的估计，每次观测 $(s, a, r, s')$ 更新：
\[
Q(s,a) \;\leftarrow\; Q(s,a) + \alpha \underbrace{\bigl[r + \gamma \max_{a'} Q(s',a') - Q(s,a)\bigr]}_{\text{TD 误差（Temporal Difference Error）}}
\]

\begin{itemize}
    \item 括号里是\textbf{时序差分误差}（TD error）：当前估计与 Bellman 方程右端之差
    \item $\alpha$ 是学习率（步长），类比随机梯度下降
    \item 无需知道 $P$，直接用样本 $s'$ 近似期望
    \item 收敛性：在一定条件下（每个 $(s,a)$ 访问无穷次，学习率满足 Robbins-Monro 条件）收敛到 $Q^*$
\end{itemize}

\textbf{策略梯度}（Policy Gradient，直接优化策略参数）

参数化策略 $\pi_\theta(a|s)$（如神经网络），直接对参数 $\theta$ 做梯度上升：
\[
\nabla_\theta J(\theta) = \mathbb{E}_{\pi_\theta} \left[ \nabla_\theta \log \pi_\theta(a|s) \cdot Q^{\pi_\theta}(s,a) \right]
\]

\begin{itemize}
    \item 右侧期望用轨迹采样估计（Monte Carlo 或 TD）
    \item $\nabla_\theta \log \pi_\theta(a|s)$：对数策略梯度（score function），指向增加动作 $a$ 概率的方向
    \item $Q^{\pi_\theta}(s,a)$：该动作的"权重"，好动作权重高，坏动作权重低
    \item \textbf{与 LP 对偶的联系}：策略梯度在\textbf{占用度量}（occupancy measure）空间里做的，正是 LP 对偶问题的梯度上升——两者的对象是同一个优化问题的两个侧面
\end{itemize}

\textbf{与 LP 框架的根本区别}

\begin{center}
\begin{tabular}{c|c|c}
方法 & 本质 & 优缺点 \\
\hline
值迭代 / 策略迭代 & 不动点迭代（函数空间） & 有理论保证，难以加约束 \\
LP 方法 & 全局优化问题 & 有对偶结构，可加约束，但规模受限 \\
强化学习 & 随机近似 / 梯度估计 & 可扩展，但收敛慢、方差大
\end{tabular}
\end{center}

\begin{keypoint}[为什么安全强化学习要回到 LP？]
值迭代和策略迭代是\textbf{不动点迭代}，本质在函数空间里找压缩映射的极限，\textbf{加约束极为麻烦}（约束会破坏压缩映射结构）。

LP 把 Bellman 不动点转化为\textbf{全局优化问题}，约束直接加到 LP 里——这正是\textbf{约束马尔可夫决策过程}（Constrained MDP, CMDP）和\textbf{安全强化学习}最终要借助 LP 或拉格朗日对偶框架的根本原因。
\end{keypoint}

\end{blackboard}

\section{附录：近端梯度法 $\bigstar$}
%========================================

\begin{blackboard}[问题设置：光滑 + 不可微]

许多实际优化问题的目标函数由两部分组成：
\[
\min_x \quad \underbrace{f(x)}_{\text{光滑凸（有梯度）}} + \underbrace{g(x)}_{\text{不可微凸（无梯度）}}
\]

典型的 $g$：
\begin{center}
\begin{tabular}{c|c|l}
$g(x)$ & 名称 & 用途 \\
\hline
$\lambda\|x\|_1$ & $\ell_1$ 范数 & 稀疏回归（LASSO）、特征选择 \\
$\mathbf{1}_C(x)$ & 指示函数 & 投影到凸集（等式 / 不等式约束） \\
$\lambda\|X\|_*$ & 核范数 & 矩阵补全（低秩约束） \\
$\lambda\|\nabla x\|_1$ & Total Variation & 图像去噪（保边缘）
\end{tabular}
\end{center}

\textbf{为什么不能直接梯度下降？}$g$ 不可微，$\nabla g$ 不存在，无法直接写出迭代格式。

\end{blackboard}

\begin{blackboard}[推导：从二次近似出发]

\textbf{普通梯度下降的本质}：每步最小化 $f$ 在当前点 $x^{(t)}$ 处的二次上界：
\[
x^{(t+1)} = \arg\min_x \; \underbrace{f(x^{(t)}) + \nabla f(x^{(t)})^\top(x - x^{(t)}) + \frac{L}{2}\|x - x^{(t)}\|^2}_{\text{$f$ 的二次近似上界（$L$-光滑保证）}}
\]

化简后就是 $x^{(t+1)} = x^{(t)} - \frac{1}{L}\nabla f(x^{(t)})$。

\textbf{加入不可微项 $g$}：保持 $g$ 原样，只对 $f$ 做二次近似：
\begin{align*}
x^{(t+1)} &= \arg\min_x \;\; g(x) + f(x^{(t)}) + \nabla f(x^{(t)})^\top(x - x^{(t)}) + \frac{L}{2}\|x - x^{(t)}\|^2 \\
&= \arg\min_x \;\; g(x) + \frac{L}{2}\left\|x - \underbrace{\left(x^{(t)} - \frac{1}{L}\nabla f(x^{(t)})\right)}_{z^{(t)}:\ \text{梯度步之后的点}}\right\|^2
\end{align*}

（常数项 $f(x^{(t)}) - \frac{1}{2L}\|\nabla f(x^{(t)})\|^2$ 与 $x$ 无关，已略去。）

\textbf{分两步理解}：
\begin{enumerate}
    \item \textbf{梯度步}：对光滑部分 $f$ 做一步梯度下降，得到 $z^{(t)} = x^{(t)} - \frac{1}{L}\nabla f(x^{(t)})$
    \item \textbf{近端步}：把 $z^{(t)}$ "投影"回让 $g$ 小的区域，得到 $x^{(t+1)}$
\end{enumerate}

\end{blackboard}

\begin{blackboard}[近端算子的定义]

\begin{definition}[近端算子]
函数 $g$ 的近端算子（proximal operator）：
\[
\operatorname{prox}_{\frac{1}{L}g}(z) = \arg\min_x \;\; g(x) + \frac{L}{2}\|x - z\|^2
\]
\end{definition}

\textbf{含义}：找一个既让 $g$ 小（第一项）、又离 $z$ 不远（第二项）的点；$1/L$ 控制两者的权衡。

\textbf{关键性质}：
\begin{itemize}
    \item 当 $g$ 是凸函数时，近端算子有\textbf{唯一解}（严格凸的 $\|x-z\|^2$ 确保唯一性）
    \item 近端算子是\textbf{非扩张映射}：$\|\operatorname{prox}(z) - \operatorname{prox}(z')\| \leq \|z - z'\|$
    \item 当 $g = 0$ 时，$\operatorname{prox}(z) = z$（退化为恒等映射）
\end{itemize}

\textbf{完整算法（近端梯度法，PGM）}：
\[
\boxed{z^{(t)} = x^{(t)} - \frac{1}{L}\nabla f(x^{(t)}), \qquad x^{(t+1)} = \operatorname{prox}_{\frac{1}{L}g}\!\left(z^{(t)}\right)}
\]

\end{blackboard}

\begin{blackboard}[三个最重要的近端算子]

\textbf{1. $g(x) = \lambda\|x\|_1$（LASSO / 稀疏约束）}

逐分量求解：对每个分量 $\min_{x_i} \lambda|x_i| + \frac{L}{2}(x_i - z_i)^2$，得\textbf{软阈值算子}：
\[
\operatorname{prox}(z_i) = \operatorname{sign}(z_i)\cdot\max\!\left(|z_i| - \frac{\lambda}{L},\; 0\right)
\]

\begin{center}
\begin{tikzpicture}[scale=0.85]
    \draw[->] (-3.2, 0) -- (3.2, 0) node[right] {$z_i$};
    \draw[->] (0, -2.2) -- (0, 2.2) node[above] {$\operatorname{prox}(z_i)$};
    % 恒等线（虚线）
    \draw[dashed, gray] (-3,-3) -- (3,3);
    % 软阈值曲线
    \def\lam{0.8}
    \draw[myblue, very thick] (-3, {-3+\lam}) -- (-\lam, 0) -- (\lam, 0) -- (3, {3-\lam});
    % 标注
    \draw[<->] (0, -0.35) -- (\lam, -0.35) node[midway, below, font=\small] {$\lambda/L$};
    \draw[<->] (0, 0.35) -- (-\lam, 0.35) node[midway, above, font=\small] {$\lambda/L$};
    \node[myblue, font=\small] at (2.5, 1.2) {软阈值};
    \node[gray, font=\small] at (2.5, 2.8) {恒等};
\end{tikzpicture}
\end{center}

绝对值 $< \lambda/L$ 的分量\textbf{精确置零}（稀疏性的来源！），其余向零收缩 $\lambda/L$。

\vspace{0.5em}
\textbf{2. $g(x) = \mathbf{1}_C(x)$（凸集指示函数）}

$\mathbf{1}_C(x) = 0$（若 $x \in C$）或 $+\infty$（否则）。近端算子退化为\textbf{投影}：
\[
\operatorname{prox}(z) = \operatorname{Proj}_C(z) = \arg\min_{x \in C} \|x - z\|^2
\]

近端梯度法退化为\textbf{投影梯度下降（PGD）}：先做梯度步，再投影回可行域。

\vspace{0.5em}
\textbf{3. $g(X) = \lambda\|X\|_*$（核范数，矩阵补全）}

设 $Z = U\,\Sigma V^\top$（SVD），近端算子为\textbf{奇异值软阈值（SVT）}：
\[
\operatorname{prox}(Z) = U\;\operatorname{diag}\!\left(\max\!\left(\sigma_i - \frac{\lambda}{L},\; 0\right)\right) V^\top
\]

对奇异值做软阈值——小奇异值置零，降低矩阵秩，推动低秩结构。

\end{blackboard}

\begin{blackboard}[近端算子为什么"不用解优化"？]

\textbf{澄清一个常见误解}：近端算子本身确实是个优化问题：
\[
\operatorname{prox}_{\alpha g}(z) = \arg\min_x \;\; g(x) + \frac{1}{2\alpha}\|x - z\|^2
\]
并没有回避优化。说"不用解优化"的意思是：对常见的 $g$，这个问题有\textbf{闭合形式解}，直接套公式 $O(n)$ 算完，\textbf{无需迭代}。

\begin{center}
\begin{tabular}{c|c|c}
$g(x)$ & $\operatorname{prox}_{\alpha g}(z)$ 的闭合形式 & 计算量 \\
\hline
$\lambda\|x\|_1$ & 软阈值，逐元素操作 & $O(n)$ \\
$\mathbf{1}_C(x)$，$C$ 是半空间 & 一次内积 + 截断 & $O(n)$ \\
$\mathbf{1}_C(x)$，$C$ 是球 & 归一化 & $O(n)$ \\
$\lambda\|x\|_2$（group lasso） & 向量软阈值 & $O(n)$ \\
$\lambda\|X\|_*$ & SVD + 奇异值软阈值 & $O(n^3)$ \\
\end{tabular}
\end{center}

\end{blackboard}

\begin{blackboard}[$\ell_1$ 近端算子闭合解的推导]

标量情形（$n=1$，$g(x) = \lambda|x|$）：
\[
\min_x \;\; \lambda|x| + \frac{1}{2\alpha}(x - z)^2
\]

$\lambda|x|$ 在 $x=0$ 处不可微，用\textbf{次梯度最优性条件}（$0 \in \partial F(x)$）：
\[
\frac{x - z}{\alpha} + \lambda \cdot \partial|x| \ni 0
\]

分三种情况讨论：

\begin{enumerate}
    \item \textbf{若 $x > 0$}：$\partial|x| = \{1\}$，条件变为 $\frac{x-z}{\alpha} + \lambda = 0$，解得 $x = z - \alpha\lambda$，要求 $z > \alpha\lambda$。
    \item \textbf{若 $x < 0$}：$\partial|x| = \{-1\}$，条件变为 $\frac{x-z}{\alpha} - \lambda = 0$，解得 $x = z + \alpha\lambda$，要求 $z < -\alpha\lambda$。
    \item \textbf{若 $x = 0$}：$\partial|x| = [-1,1]$，条件为 $\frac{-z}{\alpha} \in [-\lambda, \lambda]$，即 $|z| \leq \alpha\lambda$。
\end{enumerate}

三种情况合并，得\textbf{软阈值公式}（无需任何迭代）：
\[
\operatorname{prox}_{\alpha\lambda|\cdot|}(z) = \operatorname{sign}(z)\cdot\max(|z| - \alpha\lambda,\; 0)
\]

\textbf{向量推广}：$g(x) = \lambda\|x\|_1 = \sum_i\lambda|x_i|$ 各分量独立，对每维用上式，$O(n)$ 完成。

\textbf{为什么 $\ell_1$ 产生精确零而 $\ell_2$ 不能？}

对 $g(x) = \lambda x^2$（$\ell_2$），近端算子为 $\operatorname{prox}(z) = \frac{z}{1+2\alpha\lambda}$——对任意 $z \neq 0$ 输出均非零，只是缩小，永远无法置零。$\ell_1$ 的"死区" $|z| \leq \alpha\lambda$ 使得该区间内的分量被\textbf{精确拉到零}，这是稀疏性的几何根源。

\end{blackboard}

\begin{blackboard}[当近端算子没有闭合解时：ADMM]

$g$ 结构复杂时，近端算子本身也需要迭代，近端梯度法优势消失：

\begin{itemize}
    \item $g(x) = \log(1 + e^x)$（logistic 损失）：$\operatorname{prox}$ 无闭合解
    \item $g(x) = \|Ax\|_1$，$A$ 非对角：各分量被 $A$ 耦合，不可分离
    \item 一般凸集的指示函数（如 $X \succeq 0$ 且 $\mathcal{A}(X) = b$）：投影无公式
\end{itemize}

\textbf{解法：ADMM（交替方向乘子法）}——引入辅助变量把 $f$ 和 $g$ 拆开：
\[
\min_{x,\, y} \;\; f(x) + g(y) \quad \text{s.t.} \quad x = y
\]

三步交替更新（$\rho > 0$ 为惩罚参数）：
\begin{align*}
x^{(t+1)} &= \arg\min_x \;\; f(x) + \frac{\rho}{2}\|x - y^{(t)} + u^{(t)}\|^2 \quad \text{（只含 $f$）}\\
y^{(t+1)} &= \operatorname{prox}_{\frac{1}{\rho}g}\!\left(x^{(t+1)} + u^{(t)}\right) \quad \text{（只含 $g$，用近端算子）}\\
u^{(t+1)} &= u^{(t)} + x^{(t+1)} - y^{(t+1)} \quad \text{（对偶变量更新，$O(n)$）}
\end{align*}

只要 $f$ 的子问题和 $g$ 的近端算子各自易于求解，整体就高效。

\textbf{具体例子：LASSO 的 ADMM 分解}

原问题 $\min_x \|Ax-b\|^2 + \lambda\|x\|_1$，令 $y = x$：

\begin{itemize}
    \item \textbf{$x$-子问题}：$\min_x \|Ax-b\|^2 + \frac{\rho}{2}\|x - y^{(t)} + u^{(t)}\|^2$

    这是\textbf{岭回归}，闭合解 $x = (2A^\top A + \rho I)^{-1}(2A^\top b + \rho(y^{(t)} - u^{(t)}))$。

    矩阵 $(2A^\top A + \rho I)$ \textbf{预先分解一次}（Cholesky，$O(n^3)$），之后每步只需回代 $O(n^2)$。

    \item \textbf{$y$-子问题}：$\min_y \lambda\|y\|_1 + \frac{\rho}{2}\|y - (x^{(t+1)} + u^{(t)})\|^2$

    这是 $\ell_1$ 的近端算子，直接\textbf{软阈值} $O(n)$：$y_i = \operatorname{sign}(v_i)\max(|v_i| - \lambda/\rho,\, 0)$，其中 $v = x^{(t+1)} + u^{(t)}$。

    \item \textbf{$u$-更新}：$u \leftarrow u + x - y$，$O(n)$。
\end{itemize}

每次迭代代价：矩阵向量乘 $O(mn)$（利用预分解）+ 软阈值 $O(n)$。与直接近端梯度法（每步也需 $O(mn)$ 计算 $A^\top(Ax-b)$）代价相当，但 ADMM 把"难的耦合"藏进了预分解，适合 $A$ 结构特殊（稀疏、Toeplitz 等）的情形。

\end{blackboard}

\begin{blackboard}[收敛率分析]

\begin{theorem}[近端梯度法收敛率]
$f$ 是 $L$-光滑凸函数，$g$ 是凸函数，步长 $\alpha = 1/L$，则：
\[
f(x^{(t)}) + g(x^{(t)}) - (f^* + g^*) \leq \frac{L\|x^{(0)} - x^*\|^2}{2t} = O\!\left(\frac{1}{t}\right)
\]
\end{theorem}

与普通梯度下降（只有 $f$，$g=0$）\textbf{完全相同的 $O(1/t)$ 收敛率}，但处理了不可微项。

\textbf{加速版（FISTA，Beck \& Teboulle 2009）}：加入动量（Nesterov 加速）：
\[
y^{(t)} = x^{(t)} + \frac{t-1}{t+2}\bigl(x^{(t)} - x^{(t-1)}\bigr)
\]
\[
x^{(t+1)} = \operatorname{prox}_{\frac{1}{L}g}\!\left(y^{(t)} - \frac{1}{L}\nabla f(y^{(t)})\right)
\]

收敛率提升到 $O(1/t^2)$，是一阶方法的理论最优（在凸 + $L$-光滑条件下）。

\begin{center}
\begin{tabular}{c|c|c}
算法 & 收敛率 & 每步代价 \\
\hline
梯度下降（$g=0$） & $O(1/t)$ & $O(n)$ 梯度 \\
近端梯度法（PGM） & $O(1/t)$ & $O(n)$ 梯度 + $\operatorname{prox}$ \\
FISTA & $O(1/t^2)$ & $O(n)$ 梯度 + $\operatorname{prox}$（同 PGM） \\
\end{tabular}
\end{center}

动量带来 $O(1/t^2)$ 而每步代价与 PGM 完全相同——免费的加速！

\end{blackboard}

\begin{blackboard}[三个具体应用]

\textbf{1. LASSO（稀疏回归）}
\[
\min_x \; \underbrace{\|Ax - b\|^2}_{f,\ \text{光滑}} + \underbrace{\lambda\|x\|_1}_{g,\ \text{不可微}}
\]
$\nabla f(x) = 2A^\top(Ax - b)$，近端算子为软阈值。每步：梯度步 + 软阈值。

$\ell_1$ 惩罚把小系数压到\textbf{精确的零}（而不是趋近零），天然产生稀疏解。这是压缩感知、基因选择的核心算法。

\vspace{0.5em}
\textbf{2. 矩阵补全（Netflix 问题）}
\[
\min_X \; \underbrace{\sum_{(i,j)\in\Omega}(X_{ij} - M_{ij})^2}_{f} + \underbrace{\lambda\|X\|_*}_{g}
\]
近端算子为奇异值软阈值（SVT）。每步：梯度步（仅在观测位置计算残差）+ SVT（压低矩阵秩）。

\vspace{0.5em}
\textbf{3. 图像去噪（Total Variation）}
\[
\min_x \; \underbrace{\|x - b\|^2}_{f} + \underbrace{\lambda\|\nabla x\|_1}_{g}
\]
$g$ 是图像梯度的 $\ell_1$ 范数，惩罚不连续性但容忍边缘跳变。近端算子有闭合形式（Chambolle 算法），能\textbf{保留边缘、去除噪声}。

\end{blackboard}

\begin{insight}[近端梯度法的核心思想]

近端梯度法的精髓在于\textbf{分而治之}：

\begin{itemize}
    \item 光滑部分 $f$：用梯度（一阶信息）处理——廉价、可并行
    \item 不可微部分 $g$：用近端算子精确处理——利用 $g$ 的特殊结构
\end{itemize}

只要能高效计算 $\operatorname{prox}_g$，就能处理任意的 $f + g$ 形式。

\textbf{近端算子 vs. 投影的关系}：投影梯度下降是近端梯度法当 $g = \mathbf{1}_C$ 时的特例；核范数软阈值是矩阵版本的"投影"到低秩约束的松弛。三个例子本质上都在做同一件事：把"走了一步梯度"的中间点拉回到满足某种结构的地方。

\end{insight}
